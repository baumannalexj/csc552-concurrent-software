001package ajeffrey.teaching.jack;
002
003import ajeffrey.teaching.debug.Debug;
004import ajeffrey.teaching.io.DevNull;
005
006import java.io.PrintWriter;
007
008/**
009 * A multi-threaded implementation of the business logic for the Jack
010 * application.  This version uses Thread.suspend/resume to handle the
011 * suspend and resume calls, which works, but is deprecated.
012 * @version 1.0.1
013 * @author Alan Jeffrey
014 */
015public interface SuspendResumeLogic {
016
017    public static LogicFactory factory = new SuspendResumeLogicFactoryImpl ();
018
019}
020
021class SuspendResumeLogicFactoryImpl implements LogicFactory {
022    
023    public Logic build () { return new SuspendResumeLogicImpl (); }
024
025}
026
027class SuspendResumeLogicImpl implements Logic, Runnable {
028
029    protected final Thread thread = new Thread (this);
030
031    protected PrintWriter out = DevNull.printWriter;
032
033    protected final String message = 
034        "\nAll work and no play makes Jack a dull boy.";
035
036    protected int offset = 0;
037
038    protected void printChar () {
039        offset = (offset + 1) % (message.length ());
040        char c = message.charAt (offset);
041        Debug.out.println ("SuspendResumeLogic.printChar (): Printing " + c);
042        out.print (c);
043        out.flush ();
044    }
045
046    public void setPrintWriter (final PrintWriter out) {
047        Debug.out.println ("SuspendResumeLogic.setPrintWriter (): Starting");
048        this.out = out;
049        Debug.out.println ("SuspendResumeLogic.setPrintWriter (): Returning");
050    }
051
052    public void suspend () {
053        Debug.out.println ("SuspendResumeLogic.suspend (): Starting");
054        thread.suspend ();
055        Debug.out.println ("SuspendResumeLogic.suspend (): Returning");
056    }
057
058    public void resume () {
059        Debug.out.println ("SuspendResumeLogic.resume (): Starting");
060        thread.resume ();
061        Debug.out.println ("SuspendResumeLogic.resume (): Returning");
062    }
063
064    public void run () { 
065        Debug.out.println ("SuspendResumeLogic.run (): Starting");
066        try {
067            while (true) {
068                Thread.sleep (200);
069                printChar ();
070            }
071        } catch (InterruptedException ex) {
072            Debug.out.println 
073                ("SuspendResumeLogic.run (): Caught exception " + ex);
074        }
075        Debug.out.println ("SuspendResumeLogic.run (): Returning");
076    }
077
078    public void start () {
079        Debug.out.println ("SuspendResumeLogic.start (): Starting");
080        thread.start ();
081        Debug.out.println ("SuspendResumeLogic.start (): Returning");
082    }
083
084}
