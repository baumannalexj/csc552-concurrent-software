001package ajeffrey.teaching.jack;
002
003import java.io.PrintWriter;
004
005import java.awt.event.ActionListener;
006import java.awt.event.ActionEvent;
007
008import java.awt.Button;
009import java.awt.Frame;
010import java.awt.Panel;
011import java.awt.TextArea;
012
013import java.awt.BorderLayout;
014
015import ajeffrey.teaching.io.TextAreaIO;
016import ajeffrey.teaching.debug.Debug;
017
018/**
019 * A GUI for the Jack application.
020 * @version 1.0.1
021 * @author Alan Jeffrey
022 * @see GUIFactory
023 */
024public interface GUI {
025
026    /**
027     * Start the GUI.
028     */
029    public void start ();
030
031    /**
032     * A factory for building GUIs.
033     */
034    public static GUIFactory factory = new GUIFactoryImpl ();
035
036}
037
038class GUIFactoryImpl implements GUIFactory {
039    
040    public GUI build (final LogicFactory logicFactory) { 
041        return new GUIImpl (logicFactory); 
042    }
043
044}
045
046class GUIImpl implements GUI {
047
048    // Get round a bug with final fields in JDK1.2
049    // protected final Logic logic;
050    // protected final LogicFactory logicFactory;
051    protected Logic logic;
052    protected LogicFactory logicFactory;
053
054    protected final TextArea textArea = new TextArea ();
055    protected final Button suspendButton = new Button ("Suspend");
056    protected final Button resumeButton = new Button ("Resume");
057    protected final Button newButton = new Button ("New");
058    protected final Button quitButton = new Button ("Quit");
059    protected final Panel buttonPanel = new Panel ();
060    protected final Frame mainFrame = new Frame ();
061
062    protected final ActionListener suspendActionListener =
063        new ActionListener () { public void actionPerformed (ActionEvent ev) {
064            Debug.out.println ("suspendActionListener: Starting");
065            logic.suspend ();
066            Debug.out.println ("suspendActionListener: Returning");
067        } };
068
069    protected final ActionListener resumeActionListener =
070        new ActionListener () { public void actionPerformed (ActionEvent ev) {
071            Debug.out.println ("resumeActionListener: Starting");
072            logic.resume ();
073            Debug.out.println ("resumeActionListener: Returning");
074        } };
075
076    protected final ActionListener newActionListener =
077        new ActionListener () { public void actionPerformed (ActionEvent ev) {
078            Debug.out.println ("newActionListener: Starting");
079            Debug.out.println ("newActionListener: Creating...");
080            final GUI newGUI = new GUIImpl (logicFactory);
081            Debug.out.println ("newActionListener: Starting...");
082            newGUI.start ();
083            Debug.out.println ("newActionListener: Returning");
084        } };
085
086    protected final ActionListener quitActionListener =
087        new ActionListener () { public void actionPerformed (ActionEvent ev) {
088            Debug.out.println ("quitActionListener: Exiting!");
089            System.exit (0);
090        } };
091
092    protected final PrintWriter textAreaPrintWriter
093        = TextAreaIO.singleton.buildPrintWriter (textArea);
094
095    GUIImpl (final LogicFactory logicFactory) {
096        this.logicFactory = logicFactory;
097        this.logic = logicFactory.build ();
098        logic.setPrintWriter (textAreaPrintWriter);
099        Debug.out.println ("GUIImpl: Built");
100    }
101
102    public void start () {
103        Debug.out.println ("GUIImpl.start: Starting");
104        Debug.out.println ("GUIImpl.start: Customizing components");
105        mainFrame.setSize (300,300);
106        suspendButton.addActionListener (suspendActionListener);
107        resumeButton.addActionListener (resumeActionListener);
108        newButton.addActionListener (newActionListener);
109        quitButton.addActionListener (quitActionListener);
110        Debug.out.println ("GUIImpl.start: Adding components");
111        buttonPanel.add (suspendButton);
112        buttonPanel.add (resumeButton);
113        buttonPanel.add (newButton);
114        buttonPanel.add (quitButton);
115        mainFrame.add (textArea);
116        mainFrame.add (BorderLayout.SOUTH, buttonPanel);
117        Debug.out.println ("GUIImpl.start: Showing frame");
118        mainFrame.setVisible (true);
119        Debug.out.println ("GUIImpl.start: Starting business logic");
120        logic.start ();
121        Debug.out.println ("GUIImpl.start: Returning");
122    }
123
124}
