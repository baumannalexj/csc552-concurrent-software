001package ajeffrey.teaching.size.client;
002
003import ajeffrey.teaching.size.event.SizeRequestSerializer;
004import ajeffrey.teaching.size.event.SizeResponseParser;
005import ajeffrey.teaching.size.event.SizeResponseHandler;
006import ajeffrey.teaching.size.event.ParseException;
007
008import ajeffrey.teaching.debug.Debug;
009
010import java.net.Socket;
011
012import java.io.OutputStream;
013import java.io.IOException;
014import java.io.FileNotFoundException;
015
016/**
017 * The business logic for a client which finds the size of files on
018 * a matching server.
019 * @author Alan Jeffrey
020 * @version 1.0.1
021 */
022public interface Client {
023
024    /**
025     * Start the client.
026     * @exception IOException thrown when there is network trouble
027     */
028    public void start () throws IOException;
029
030    /**
031     * Stop the client.
032     * @exception IOException thrown when there is network trouble
033     */
034    public void stop () throws IOException;
035
036    /**
037     * Get the size of a file.
038     * @param fileName the name of the file
039     * @return the size of the file
040     * @exception IOException thrown when there is network trouble
041     * @exception FileNotFoundException thown when the file does not exist
042     */
043    public long getSize (String fileName) throws IOException, FileNotFoundException;
044
045    /**
046     * A factory for building clients
047     */
048    public static ClientFactory factory = new ClientFactoryImpl ();
049
050}
051
052class ClientFactoryImpl implements ClientFactory {
053
054    public Client build (final String hostName, final int port) {
055        return new ClientImpl (hostName, port);
056    }
057
058}
059
060class ClientImpl implements Client, SizeResponseHandler {
061
062    protected final String hostName;
063    protected final int port;
064    protected Socket socket;
065    protected SizeRequestSerializer serializer;
066    protected SizeResponseParser parser;
067
068    protected ClientImpl (final String hostName, final int port) {
069        this.hostName = hostName;
070        this.port = port;
071        Debug.out.println ("ClientImpl: built");
072    }
073
074    public void start () throws IOException {
075        Debug.out.println ("ClientImpl.start: Starting");
076        Debug.out.println ("ClientImpl.start: Opening socket to " + hostName + ":" + port);
077        socket = new Socket (hostName, port);
078        Debug.out.println ("ClientImpl.start: Building serializer");
079        serializer = SizeRequestSerializer.factory.build (socket.getOutputStream ());
080        Debug.out.println ("ClientImpl.start: Building parser");
081        parser = SizeResponseParser.factory.build (socket.getInputStream ());
082        Debug.out.println ("ClientImpl.start: Adding handler");
083        parser.addSizeResponseHandler (this);
084        Debug.out.println ("ClientImpl.start: Returning");
085    }
086
087    public void stop () throws IOException {
088        Debug.out.println ("ClientImpl.stop: Starting");
089        serializer.handleQuitRequest ();
090        socket.close ();
091        Debug.out.println ("ClientImpl.stop: Returning");
092    }
093
094    protected long size;
095
096    public long getSize (final String fileName) throws FileNotFoundException, IOException {
097        Debug.out.println ("ClientImpl.getSize: Starting");
098        try {
099            Debug.out.println ("ClientImpl.getSize: Sending size request for " + fileName);
100            serializer.handleSizeRequest (fileName);
101            Debug.out.println ("ClientImpl.getSize: Waiting for response");
102            parser.parseSizeResponse ();
103            Debug.out.println ("ClientImpl.getSize: Got response");
104            if (size < 0) {
105                Debug.out.println ("ClientImpl.getSize: Oh dear, file not found");
106                throw new FileNotFoundException ();
107            } else {
108                Debug.out.println ("ClientImpl.getSize: File found, size = " + size);
109                return size;
110            }
111        } catch (final ParseException ex) {
112            Debug.out.println ("ClientImpl.getSize: Caught " + ex);
113            throw new IOException ("Parse error: " + ex);
114        } finally {
115            Debug.out.println ("ClientImpl.getSize: Returning");
116        }
117    }
118
119    public void handleFoundResponse (final String fileName, final long size) throws IOException {
120        Debug.out.println ("ClientImpl.handleFoundResponse: Starting");
121        Debug.out.println ("ClientImpl.handleFoundResponse: Setting size = " + size);
122        this.size = size;
123        Debug.out.println ("ClientImpl.handleFoundResponse: Returning");
124    }
125
126    public void handleNotFoundResponse () throws IOException {
127        Debug.out.println ("ClientImpl.handleNotFoundResponse: Starting");
128        this.size = -1;
129        Debug.out.println ("ClientImpl.handleNotFoundResponse: Returning");
130    }
131
132    public void handleOKResponse () throws IOException {
133        Debug.out.println ("ClientImpl.handleOKResponse: Starting");
134        Debug.out.println ("ClientImpl.handleOKResponse: Returning");
135    }
136
137    public void handleEndOfStream () throws IOException {
138        Debug.out.println ("ClientImpl.handleEndOfStream: Starting");
139        throw new IOException ("Stream closed");
140    }
141
142}
