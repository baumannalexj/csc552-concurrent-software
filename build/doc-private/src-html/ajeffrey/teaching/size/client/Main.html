001package ajeffrey.teaching.size.client;
002
003import java.io.IOException;
004import java.io.FileNotFoundException;
005
006import ajeffrey.teaching.debug.Debug;
007
008/**
009 * A client which finds the size of files given by a size server.
010 * It takes a server name, a port and a list of files as arguments.
011 * @author Alan Jeffrey
012 * @version 1.0.1
013 */
014public class Main {
015
016    public static void main (String[] args) throws IOException {
017        Debug.out.addPrintStream (System.err);
018        final String hostName = args[0];
019        final int port = Integer.parseInt (args[1]);
020        Debug.out.println ("Main.main: Building client");
021        final Client client = Client.factory.build (hostName, port);
022        Debug.out.println ("Main.main: Starting client");
023        client.start ();
024        for (int i = 2; i<args.length; i++) {
025            Debug.out.println ("Main.main: Finding size of " + args[i]);
026            try {
027                long size = client.getSize (args[i]);
028                System.out.println ("File " + args[i] + " is " + size + " bytes");
029            } catch (final FileNotFoundException ex) {
030                System.out.println ("File " + args[i] + " does not exist");
031            }
032        }
033        Debug.out.println ("Main.main: Stopping client");
034        client.stop ();
035        Debug.out.println ("Main.main: Returning");
036    }
037
038}
