001package ajeffrey.teaching.size.event;
002
003import ajeffrey.teaching.debug.Debug;
004
005import java.io.OutputStream;
006import java.io.OutputStreamWriter;
007import java.io.IOException;
008import java.io.UnsupportedEncodingException;
009
010import com.macfaq.io.SafePrintWriter;
011
012/**
013 * A handler which sends size responses to an output stream.
014 * @author Alan Jeffrey
015 * @version 1.0.1
016 */
017public interface SizeResponseSerializer extends SizeResponseHandler {
018
019    /**
020     * A factory for building new response serializers.
021     */
022    public static final SizeResponseSerializerFactory factory =
023        new SizeResponseSerializerFactoryImpl ();
024
025}
026
027class SizeResponseSerializerFactoryImpl implements SizeResponseSerializerFactory {
028
029    public SizeResponseSerializer build (final OutputStream out) {
030        return new SizeResponseSerializerImpl (out);
031    }
032
033}
034
035class SizeResponseSerializerImpl implements SizeResponseSerializer {
036
037    protected final SafePrintWriter out;
038
039    protected SizeResponseSerializerImpl (final OutputStream out) {
040        try {
041            this.out = new SafePrintWriter (new OutputStreamWriter (out, "UTF-8"), true, "\n");
042        } catch (final UnsupportedEncodingException ex) {
043            System.err.println ("This shouldn't happen! " + ex);
044            System.exit (1);
045            throw new RuntimeException ("A stupid exception to get past Java's control flow analysis");
046        }
047        Debug.out.println ("SizeResponseSerializerImpl: built");
048    }
049
050    public void handleFoundResponse (final String fileName, final long size) throws IOException {
051        Debug.out.println 
052            ("SizeResponseSerializerImpl.handleFoundResponse: sending FOUND " 
053             + fileName + " " + size + " BYTES");
054        out.println ("FOUND " + fileName + " " + size + " BYTES");
055    }
056
057    public void handleNotFoundResponse () throws IOException {
058        Debug.out.println 
059            ("SizeResponseSerializerImpl.handleNotFoundResponse: sending NOT FOUND");
060        out.println ("NOT FOUND");
061    }
062
063    public void handleOKResponse () throws IOException {
064        Debug.out.println 
065            ("SizeResponseSerializerImpl.handleOKResponse: sending OK");
066        out.println ("OK");
067    }
068
069    public void handleEndOfStream () throws IOException {
070        Debug.out.println 
071            ("SizeResponseSerializerImpl.handleEndOfStream: closing stream");
072        out.close ();
073    }
074
075}
