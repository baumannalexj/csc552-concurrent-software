001package ajeffrey.teaching.size.event;
002
003import java.io.IOException;
004
005/**
006 * An event handler for a SIZE request (either SIZE or QUIT).
007 * @author Alan Jeffrey
008 * @version 1.0.1
009 */
010public interface SizeRequestHandler {
011
012    /**
013     * Handle a SIZE request
014     * @param fileName the name of the file
015     */
016    public void handleSizeRequest (String fileName) throws IOException;
017
018    /**
019     * Handle a QUIT request
020     */
021    public void handleQuitRequest () throws IOException;
022
023}
