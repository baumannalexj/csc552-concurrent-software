001package ajeffrey.teaching.date;
002
003import ajeffrey.teaching.debug.Debug;
004
005import javax.swing.JFrame;
006import javax.swing.JPanel;
007import javax.swing.JLabel;
008import javax.swing.JTextField;
009import javax.swing.JButton;
010import javax.swing.JOptionPane;
011
012import java.awt.BorderLayout;
013
014import java.awt.event.ActionListener;
015import java.awt.event.ActionEvent;
016
017public interface View {
018
019    public void start ();
020
021    public static ViewFactory factory = new ViewFactoryImpl ();
022
023}
024
025class ViewFactoryImpl implements ViewFactory {
026
027    public View build (Model model) { return new ViewImpl (model); }
028
029}
030
031class ViewImpl implements View {
032
033    final Model model;
034
035    final JFrame frame = new JFrame ("Date");
036
037    final JLabel connectLabel = new JLabel ("Date Server:");
038    final JTextField connectField = new JTextField ();
039    final JButton connectButton = new JButton ("Connect");
040
041    final JLabel dateLabel = new JLabel ("--- --- -- --:--:-- --- ----", JLabel.CENTER);
042
043    final JButton dateButton = new JButton ("Refresh");
044    final JButton quitButton = new JButton ("Quit");
045
046    final ActionListener connectListener = new ActionListener () {
047            public void actionPerformed (ActionEvent ev) {
048                try {
049                    Debug.out.println ("Connect button pressed");
050                    model.connect (connectField.getText ());
051                    Debug.out.println ("Connect button finished");
052                    JOptionPane.showMessageDialog (frame, "Connection Succeeded");
053                } catch (Exception ex) {
054                    JOptionPane.showMessageDialog (frame, ex.toString (), "Connection Failed", JOptionPane.ERROR_MESSAGE);
055                }
056            }
057        };
058
059    final ActionListener dateListener = new ActionListener () {
060            public void actionPerformed (ActionEvent ev) {
061                try {
062                    Debug.out.println ("Date button pressed");
063                    dateLabel.setText (model.getDate ().toString ());
064                    Debug.out.println ("Date button finished");
065                } catch (Exception ex) {
066                    JOptionPane.showMessageDialog (frame, ex.toString (), "Refresh Failed", JOptionPane.ERROR_MESSAGE);
067                }
068            }
069        };
070
071    final ActionListener quitListener = new ActionListener () {
072            public void actionPerformed (ActionEvent ev) {
073                Debug.out.println ("Quit button pressed");
074                System.exit (0);
075            }
076        };
077
078    ViewImpl (Model model) { this.model = model; }
079
080    public void start () {
081        connectButton.addActionListener (connectListener);
082        dateButton.addActionListener (dateListener);
083        quitButton.addActionListener (quitListener);
084        final JPanel connectPanel = new JPanel (new BorderLayout ());
085        final JPanel buttonPanel = new JPanel (new BorderLayout ());
086        connectPanel.add ("West", connectLabel);
087        connectPanel.add ("Center", connectField);
088        connectPanel.add ("East", connectButton);
089        buttonPanel.add ("West", dateButton);
090        buttonPanel.add ("East", quitButton);
091        frame.getContentPane ().add ("North", connectPanel);
092        frame.getContentPane ().add ("Center", dateLabel);
093        frame.getContentPane ().add ("South", buttonPanel);
094        frame.setSize (300,100);
095        frame.setVisible (true); 
096    }
097
098}
