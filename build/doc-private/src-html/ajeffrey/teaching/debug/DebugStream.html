001package ajeffrey.teaching.debug;
002
003/**
004 * An inferface for printing debugging messages.
005 * @author Alan Jeffrey
006 * @version 1.0.3
007 */
008public interface DebugStream {
009
010    /**
011     * Prints a debugging message.
012     * @param msg the message to print
013     */
014    public void println (String msg);
015
016    /**
017     * Prints a debugging message and sets a break point.
018     * Note that this should not be done in the main GUI thread!
019     * @param msg the message to print
020     */
021    public void breakPoint (String msg);
022
023}
