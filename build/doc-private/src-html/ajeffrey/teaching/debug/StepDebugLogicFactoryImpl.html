001package ajeffrey.teaching.debug;
002
003import ajeffrey.teaching.observer.Subject;
004import ajeffrey.teaching.observer.Observer;
005
006/**
007 * The logic behind a step debug GUI.
008 * This logic includes the Gang of Four's Observer/Observed pattern.
009 * Whenever the state of the object changes, it will notify
010 * any observers which are attached to it.
011 * @author Alan Jeffrey
012 * @version v1.0.1
013 */
014public interface StepDebugLogic extends Subject {
015
016    /**
017     * If we are currently in pause mode, then block,
018     * waiting for us to enter step or run mode.
019     * Then, if we are in step mode, set the debugger 
020     * back to pause mode.
021     */
022    public void waitWhilePaused ();
023
024    /**
025     * Put the debugger in pause mode.
026     */
027    public void pauseMode ();
028
029    /**
030     * Put the debugger in run mode.
031     */
032    public void runMode ();
033
034    /**
035     * Put the debugger in step mode.
036     */
037    public void stepMode ();
038
039    /**
040     * Are we in run mode?
041     * @return true if we are currently in run mode.
042     */
043    public boolean inRunMode ();
044
045    /**
046     * Are we in step mode?
047     * @return true if we are currently in step mode.
048     */
049    public boolean inStepMode ();
050
051    /**
052     * Are we in pause mode?
053     * @return true if we are currently in pause mode.
054     */
055    public boolean inPauseMode ();
056
057    /**
058     * A string corresponding to the name of the current thread.
059     * @return the name of the current thread
060     */
061    public String getThreadName ();
062
063    /**
064     * A string corresponding to the name of the current mode.
065     * @return the name of the current mode
066     */
067    public String getModeName ();
068
069    /**
070     * A factory for building step debugger logic.
071     */
072    public static StepDebugLogicFactory factory 
073        = new StepDebugLogicFactoryImpl ();
074
075}
076
077class StepDebugLogicFactoryImpl implements StepDebugLogicFactory {
078   
079    public StepDebugLogic build (String threadName) {
080        return new StepDebugLogicImpl (threadName);
081    }
082
083}
084
085class StepDebugLogicImpl implements StepDebugLogic {
086
087    protected final int PAUSE = 0;
088    protected final int RUN = 1;
089    protected final int STEP = 2;
090
091    protected int mode = RUN;
092    protected final Object lock = new Object ();
093    protected final Subject subject = Subject.factory.build ();
094    protected final String threadName;
095
096    protected StepDebugLogicImpl (final String threadName) {
097        this.threadName = threadName;
098    }
099
100    public void waitWhilePaused () {
101        if (!inRunMode ()) {
102            if (inStepMode ()) { pauseMode (); }
103            synchronized (lock) {
104                while (inPauseMode ()) { 
105                    try {
106                        lock.wait (); 
107                    } catch (final InterruptedException ex) {
108                        Thread.currentThread ().interrupt ();
109                    }
110                }
111            }
112        }
113    }
114
115    public void pauseMode () {
116        mode = PAUSE;
117        updateObservers ();
118    }
119
120    public void runMode () {
121        mode = RUN;
122        updateObservers ();
123        synchronized (lock) { lock.notifyAll (); }
124    }
125
126    public void stepMode () {
127        mode = STEP;
128        updateObservers ();
129        synchronized (lock) { lock.notifyAll (); }
130    }
131
132    public boolean inRunMode () { return mode == RUN; }
133
134    public boolean inStepMode () { return mode == STEP; }
135
136    public boolean inPauseMode () { return mode == PAUSE; }
137
138    public String getModeName () {
139        switch (mode) {
140        case RUN: return "Running";
141        case STEP: return "Stepping";
142        case PAUSE: return "Paused";
143        default: return "BUGGY";
144        }
145    }
146
147    public String getThreadName () {
148        return threadName;
149    }
150
151    public void attach (final Observer observer) {
152        subject.attach (observer);
153    }
154
155    public void detach (final Observer observer) {
156        subject.detach (observer);
157    }
158
159    public void updateObservers () {
160        subject.updateObservers ();
161    }
162
163}
