001package ajeffrey.teaching.catalog;
002
003import ajeffrey.teaching.util.Dictionary;
004
005import java.util.StringTokenizer;
006
007public interface Catalog {
008
009    public Catalog addTitle (String title);
010    public ResultSet search (String keyword);
011
012    public static final Catalog empty = new CatalogImpl (Dictionary.empty);
013
014}
015
016class CatalogImpl implements Catalog {
017
018    final Dictionary contents;
019
020    CatalogImpl (Dictionary contents) {
021        this.contents = contents;
022    }
023
024    public Catalog addTitle (String title) {
025        final Book book = new BookImpl (title);
026        final String[] keywords = book.getKeywords ();
027        Dictionary contents = this.contents;
028        for (int i=0; i<keywords.length; i++) {
029            final ResultSetImpl results = 
030                (ResultSetImpl)(search (keywords[i]));
031            contents = contents.add (keywords[i], results.add (book));
032        }
033        return new CatalogImpl (contents);
034    }
035    
036    public ResultSet search (String keyword) {
037        final ResultSet results = (ResultSet)(contents.get (keyword));
038        if (results == null) { return ResultSetImpl.empty; }
039        else { return results; }
040    }
041
042}
043
044class ResultSetImpl implements ResultSet {
045
046    final static ResultSet empty = 
047        new ResultSetImpl (Dictionary.empty, Dictionary.empty, Dictionary.empty, 0);
048
049    final Dictionary keywords;
050    final Dictionary books;
051    final Dictionary contents;
052    final int size;
053
054    ResultSetImpl (Dictionary keywords, Dictionary books, Dictionary contents, int size) { 
055        this.keywords = keywords;
056        this.books = books;
057        this.contents = contents; 
058        this.size = size;
059    }
060
061    public int size () { return size; }
062
063    public Book[] getBooks () {
064        final Object[] bookTitles = books.getKeys ();
065        final Book[] result = new Book[bookTitles.length];
066        for (int i=0; i<bookTitles.length; i++) {
067            result[i] = (Book)(books.get (bookTitles[i]));
068        }
069        return result;
070    }
071    
072    public ResultSet refine (String keyword) {
073        if (keywords.containsKey (keyword)) { 
074            return this; 
075        } else {
076            final ResultSet result = (ResultSet)(contents.get (keyword));
077            if (result == null) {
078                return new ResultSetImpl 
079                    (keywords.add (keyword, keyword), Dictionary.empty, Dictionary.empty, 0);
080            } else { 
081                return result; 
082            }
083        }
084    }
085
086    ResultSet add (Book book) {
087        if (this.books.containsKey (book.getTitle ())) {
088            return this;
089        } else {
090            final String[] keywords = book.getKeywords ();
091            Dictionary contents = this.contents;
092            for (int i=0; i<keywords.length; i++) {
093                if (!this.keywords.containsKey (keywords[i])) {
094                    ResultSetImpl results = 
095                        (ResultSetImpl)(refine (keywords[i]));
096                    contents = contents.add (keywords[i],results.add (book));
097                }
098            }
099            return new ResultSetImpl 
100                (this.keywords, 
101                 this.books.add (book.getTitle (), book), 
102                 contents, 
103                 this.size+1);
104        }
105    }
106
107}
108
109class BookImpl implements Book {
110
111    final String title;
112    final String[] keywords;
113
114    BookImpl (String title) {
115        final StringTokenizer st = new StringTokenizer (title);
116        final String[] keywords = new String [st.countTokens ()];
117        for (int i=0; i<keywords.length; i++) {
118            keywords[i] = st.nextToken ();
119        }
120        this.title = title;
121        this.keywords = keywords;
122    }
123
124    public String getTitle () { return title; } 
125    public String[] getKeywords () { return keywords; }
126
127}
