001package ajeffrey.teaching.minisoap;
002
003public class Return implements Task {
004
005    public String resultId;
006    public Object returnResult;
007
008    public Return () { }
009
010    public Return (String resultId, Object returnResult) {
011        this.resultId = resultId;
012        this.returnResult = returnResult;
013    }
014
015    public void run (Executor exec) {
016        Result result = (Result)(Result.lookup.get (resultId));
017        result.returnResult (returnResult);
018        Result.lookup.remove (resultId);
019    }
020
021    public String toString () { return "return " + returnResult + " to " + resultId; }
022
023
024}
