001package ajeffrey.teaching.minisoap;
002
003import com.macfaq.io.SafePrintWriter;
004import ajeffrey.teaching.debug.Debug;
005import java.io.IOException;
006
007public interface RMISoapWriter extends SoapWriter {
008   
009    public static SoapWriterFactory factory = new RMISoapWriterFactoryImpl ();
010
011}
012
013class RMISoapWriterFactoryImpl extends SoapWriterFactoryImpl {
014
015    public SoapWriter build (final SafePrintWriter out) {
016        return new RMISoapWriterImpl (out);
017    }
018
019}
020
021class RMISoapWriterImpl extends SoapWriterImpl {
022
023    protected RMISoapWriterImpl (final SafePrintWriter out) {
024        super (out);
025    }
026
027    public void serialize (final Object obj) throws IOException {
028        if (obj instanceof Remote) {
029            Debug.out.println ("RMISoapWriterImpl.serialize: serializing stub instead of object!");
030            super.serialize (Stub.factory.build ((Remote)obj));
031        } else {
032            super.serialize (obj);
033        }
034    }
035
036}
