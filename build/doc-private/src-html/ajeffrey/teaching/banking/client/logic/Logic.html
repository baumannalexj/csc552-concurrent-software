001package ajeffrey.teaching.banking.client.logic;
002
003import ajeffrey.teaching.observer.Subject;
004import ajeffrey.teaching.observer.Observer;
005import ajeffrey.teaching.debug.Debug;
006
007/**
008 * The business logic for the banking client.
009 * @author Alan Jeffrey
010 * @version 1.0.2
011 */
012public interface Logic {
013
014    /**
015     * A factory for building new business logic objects.
016     */
017    public static final LogicFactory factory = new LogicFactoryImpl ();
018
019    /**
020     * A Subject for the business logic, suitable for using with the
021     * observer pattern.  Whenever the logic changes state, this subject
022     * will update all its observers.
023     * @return the Subject for this logic object.
024     */
025    public Subject getSubject ();
026
027    /**
028     * The current total credits to the account
029     * @return the total credits to the account
030     */
031    public int getCredits ();
032
033    /**
034     * The current total debits to the account
035     * @return the total debits to the account
036     */
037    public int getDebits ();
038
039    /**
040     * The current total balance of the account
041     * @return the total balance of the account
042     */
043    public int getBalance ();
044
045    /**
046     * Make a credit of the given amount.
047     * @param amount the amount to credit
048     */
049    public void credit (int amount);
050
051    /**
052     * Make a debit of the given amount.
053     * @param amount the amount to debit
054     * @exception OverdrawnException thrown if the amount is 
055     *   greater than the current balance
056     */
057    public void debit (int amount) throws OverdrawnException;
058
059    /**
060     * Make a credit of the given amount.
061     * @param amount the amount to credit
062     * @exception NumberFormatException thrown if the amount is 
063     *   not a positive number
064     */
065    public void credit (String amount) throws NumberFormatException;
066
067    /**
068     * Make a debit of the given amount.
069     * @param amount the amount to debit
070     * @exception NumberFormatException thrown if the amount is 
071     *  not a positive number
072     * @exception OverdrawnException thrown if the amount is 
073     *  greater than the current balance
074     */
075    public void debit (String amount) throws OverdrawnException, NumberFormatException;
076
077}
078
079class LogicFactoryImpl implements LogicFactory {
080
081    public Logic build () { return new LogicImpl (); }
082
083}
084
085class LogicImpl implements Logic {
086
087    protected final Subject subject = Subject.factory.build ();
088
089    protected int debits = 0;
090    protected int credits = 0;
091    protected int balance = 0;
092
093    public Subject getSubject () { return subject; }
094    public int getDebits () { return debits; }
095    public int getCredits () { return credits; }
096    public int getBalance () { return balance; }
097
098    public void credit (final String amountString)
099    throws NumberFormatException {
100        Debug.out.println 
101            ("Logic.credit: Starting");
102        Debug.out.println 
103            ("Logic.credit: Calling parseInt (" + amountString + ")");
104        final int amount = Integer.parseInt (amountString);
105        if (amount < 0) { 
106            Debug.out.println ("Logic.credit: Negative amount");
107            throw new NumberFormatException () ;
108        }
109        Debug.out.println ("Logic.credit: Calling credit (" + amount + ")");
110        credit (amount);
111        Debug.out.println ("Logic.credit: Returning");
112    }
113
114    public void debit (final String amountString)
115    throws NumberFormatException, OverdrawnException {
116        Debug.out.println 
117            ("Logic.debit: Starting");
118        Debug.out.println 
119            ("Logic.debit: Calling parseInt (" + amountString + ")");
120        final int amount = Integer.parseInt (amountString);
121        if (amount < 0) { 
122            Debug.out.println ("Logic.debit: Negative amount");
123            throw new NumberFormatException () ;
124        }
125        Debug.out.println ("Logic.debit: Calling debit (" + amount + ")");
126        debit (amount);
127        Debug.out.println ("Logic.debit: Returning");
128    }
129
130    public void credit (final int amount) {
131        Debug.out.println ("Logic.credit: Starting");
132        Debug.out.println ("Logic.credit: State was: " + this);
133        credits = credits + amount;
134        balance = balance + amount;
135        Debug.out.println ("Logic.credit: State is now: " + this);
136        subject.updateObservers ();
137        Debug.out.println ("Logic.credit: Returning");
138    }
139
140    public void debit (final int amount) 
141    throws OverdrawnException {
142        Debug.out.println ("Logic.debit: Starting");
143        if (amount > balance) { 
144            Debug.out.println 
145                ("Logic.debit: Tried to overdraw: amount = " +
146                 amount + ". balance = " + balance);
147            throw new OverdrawnException (); 
148        }
149        Debug.out.println ("Logic.debit: State was: " + this);
150        debits = debits + amount;
151        balance = balance - amount;
152        Debug.out.println ("Logic.debit: State is now: " + this);
153        Debug.out.println ("Logic.debit: calling subject.updateObservers ()");
154        subject.updateObservers ();
155        Debug.out.println ("Logic.debit: Returning");
156    }
157
158    public String toString () { 
159        return "Logic { debits = " + debits +
160            ", credits = " + credits + 
161            ", balance = " + balance + " }"; 
162    }
163
164}
