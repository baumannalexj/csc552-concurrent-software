001package ajeffrey.teaching.banking.client.gui;
002
003import ajeffrey.teaching.observer.Observer;
004import ajeffrey.teaching.banking.client.logic.Logic;
005import ajeffrey.teaching.banking.client.logic.OverdrawnException;
006import ajeffrey.teaching.debug.Debug;
007
008import java.awt.Button;
009import java.awt.Dialog;
010import java.awt.Frame;
011import java.awt.Label;
012import java.awt.Panel;
013import java.awt.TextField;
014
015import java.awt.event.ActionEvent;
016import java.awt.event.ActionListener;
017import java.awt.event.WindowAdapter;
018import java.awt.event.WindowEvent;
019import java.awt.event.WindowListener;
020
021import java.awt.GridLayout;
022
023/**
024 * The graphical user interface for the banking client.
025 * This is an instance of the Observer pattern from the 
026 * `Gang of Four' book.  It acts as an observer of the 
027 * business logic.  Whenever the business logic updates
028 * itself, it should call the <code>update ()</code>
029 * method of the GUI, so it can redraw itself.
030 * @author Alan Jeffrey
031 * @version 1.0.2
032 * @see Logic
033 */
034public interface GUI extends Runnable, Observer {
035
036    /**
037     * A factory for building new GUIs.
038     */
039    public static final GUIFactory factory = new GUIFactoryImpl ();
040
041    /**
042     * When the GUI is run, it will display itself.
043     */
044    public void run ();
045}
046
047class GUIFactoryImpl implements GUIFactory {
048
049    public GUI build (final Logic logic) { return new GUIImpl (logic); }
050
051}
052
053class GUIImpl implements GUI {
054
055    // Get round a bug with final fields in JDK1.2
056    // protected final Logic logic;
057    protected Logic logic;
058
059    protected final Frame frame = new Frame ("Banking Client");
060    protected final Panel balancePanel = new Panel ();
061    protected final Panel creditPanel = new Panel ();
062    protected final Panel debitPanel = new Panel ();
063    protected final Panel quitPanel = new Panel ();
064    protected final Button quitButton = new Button ("Quit");
065    protected final Label balanceLabel = new Label ();
066    protected final Label creditsLabel = new Label ();
067    protected final Label debitsLabel = new Label ();
068    protected final Label debitLabel = new Label ("New debit: ");
069    protected final Label creditLabel = new Label ("New credit: ");
070    protected final TextField debitField = new TextField (20);
071    protected final TextField creditField = new TextField (20);
072    protected final Button debitButton = new Button ("Debit");
073    protected final Button creditButton = new Button ("Credit");
074 
075    protected final WindowListener exitWindowListener = 
076        new WindowAdapter () {
077            public void windowClosing (WindowEvent ev) {
078                Debug.out.println ("exitWindowListener.windowClosing: Exiting");
079                System.exit (0);
080            }
081        };
082
083    protected final ActionListener exitActionListener = 
084        new ActionListener () {
085            public void actionPerformed (ActionEvent ev) {
086                Debug.out.println 
087                    ("exitActionListener.actionPerformed: Exiting");
088                System.exit (0);
089            }
090        };
091
092    protected final ActionListener creditActionListener = 
093        new ActionListener () {
094            public void actionPerformed (ActionEvent ev) {
095                Debug.out.println 
096                    ("creditActionListener.actionPerformed: Starting");
097                final String creditString = creditField.getText ();
098                try {
099                    Debug.out.println 
100                        ("creditActionListener.actionPerformed: calling " 
101                         + logic + ".credit (" + creditString + ")");
102                    logic.credit (creditString);
103                } catch (NumberFormatException ex) {
104                    Debug.out.println
105                        ("creditActionListener.actionPerformed: " 
106                         + "calling errorMessage");
107                    errorMessage 
108                        ("The amount entered: \"" + creditString 
109                         + "\" does not appear to be a number");
110                }
111                Debug.out.println 
112                    ("creditActionListener.actionPerformed: Returning");
113            }
114        };
115
116    protected final ActionListener debitActionListener = 
117        new ActionListener () {
118            public void actionPerformed (ActionEvent ev) {
119                Debug.out.println 
120                    ("debitActionListener.actionPerformed: Starting");
121                final String debitString = debitField.getText ();
122                try {
123                    Debug.out.println 
124                        ("debitActionListener.actionPerformed: calling " 
125                         + logic + ".debit (" + debitString + ")");
126                    logic.debit (debitString);
127                } catch (OverdrawnException ex) {
128                    Debug.out.println
129                        ("debitActionListener.actionPerformed: " 
130                         + "calling errorMessage");
131                    errorMessage 
132                        ("That debit would make you overdrawn.");
133                } catch (NumberFormatException ex) {
134                    Debug.out.println
135                        ("debitActionListener.actionPerformed: " 
136                         + "calling errorMessage");
137                    errorMessage 
138                        ("The amount entered: \"" + debitString 
139                         + "\" does not appear to be a number");
140                }
141                Debug.out.println 
142                    ("debitActionListener.actionPerformed: Returning");
143            }
144        };
145
146    protected void errorMessage (final String msg) {
147        Debug.out.println ("GUI.errorMessage: Starting");
148        Debug.out.println ("GUI.errorMessage: Message is " + msg);
149        Debug.out.println ("GUI.errorMessage: Preparing components");
150        final Dialog errorDialog = new Dialog (frame, "Error Message", true);
151        final Button okButton = new Button ("OK");
152        final Panel p1 = new Panel ();
153        final Panel p2 = new Panel ();
154        okButton.addActionListener (new ActionListener () {
155                public void actionPerformed (ActionEvent ev) {
156                    Debug.out.println ("GUI.errorMessage: Disposing");
157                    errorDialog.dispose ();
158                }
159            });
160        p1.add (new Label (msg));
161        p2.add (okButton);
162        errorDialog.setLayout (new GridLayout (2,1));
163        errorDialog.add (p1);
164        errorDialog.add (p2);
165        errorDialog.setSize (400,100);
166        Debug.out.println ("GUI.errorMessage: Showing error message");
167        errorDialog.show ();
168        Debug.out.println ("GUI.errorMessage: Returning");
169    }
170
171    protected GUIImpl (final Logic logic) {
172        this.logic = logic;
173        Debug.out.println ("GUIImpl: Built");
174    }
175
176    public void run () {
177        Debug.out.println ("GUIImpl.run: Starting.");
178        Debug.out.println ("GUIImpl.run: Customizing components");
179        frame.setLayout (new GridLayout (4,1));
180        Debug.out.println ("GUIImpl.run: Adding listeners");
181        frame.addWindowListener (exitWindowListener);
182        quitButton.addActionListener (exitActionListener);
183        debitButton.addActionListener (debitActionListener);
184        creditButton.addActionListener (creditActionListener);
185        Debug.out.println ("GUIImpl.run: Adding components");
186        balancePanel.add (creditsLabel);
187        balancePanel.add (debitsLabel);
188        balancePanel.add (balanceLabel);
189        debitPanel.add (debitLabel);
190        debitPanel.add (debitField);
191        debitPanel.add (debitButton);
192        creditPanel.add (creditLabel);
193        creditPanel.add (creditField);
194        creditPanel.add (creditButton);
195        quitPanel.add (quitButton);
196        frame.add (balancePanel);
197        frame.add (debitPanel);
198        frame.add (creditPanel);
199        frame.add (quitPanel);
200        Debug.out.println ("GUIImpl.run: Calling update ()");
201        update ();
202        Debug.out.println ("GUIImpl.run: Displaying frame");
203        frame.setSize (400,200);
204        frame.setVisible (true);
205        Debug.out.println ("GUIImpl.run: Returning.");
206    }
207
208    public void update () {
209        Debug.out.println ("GUIImpl.update: Starting.");
210        Debug.out.println ("GUIImpl.update: Getting values.");
211        final int credits = logic.getCredits ();
212        final int debits = logic.getDebits ();
213        final int balance = logic.getBalance ();
214        Debug.out.println ("GUIImpl.update: Updating fields.");
215        creditsLabel.setText ("Total credits: $" + credits);
216        debitsLabel.setText ("Total debits: $" + debits);
217        balanceLabel.setText ("Total balance: $" + balance);
218        Debug.out.println ("GUIImpl.update: Returning.");
219    }
220
221    public String toString () {
222        return "GUI { logic=" + logic + " }";
223    }
224
225}
