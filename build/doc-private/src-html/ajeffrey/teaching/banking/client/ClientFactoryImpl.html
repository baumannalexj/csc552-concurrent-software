001package ajeffrey.teaching.banking.client;
002
003import ajeffrey.teaching.banking.client.logic.Logic;
004import ajeffrey.teaching.banking.client.gui.GUI;
005
006import ajeffrey.teaching.debug.Debug;
007
008/**
009 * The specification of the SE&nbsp;552 banking client.
010 * The client contains:
011 * <ul>
012 * <li>A business logic component <code>Logic</code>, and</li>
013 * <li>A GUI component <code>GUI</code>.</li>
014 * </ul>
015 * Together, these define the functionality and appearance
016 * of the client.
017 * 
018 * @author Alan Jeffrey
019 * @version 1.0.1
020 * @see Logic
021 * @see GUI
022 */
023public interface Client extends Runnable {
024
025    /**
026     * A factory for building new clients.
027     */
028    public static ClientFactory factory = new ClientFactoryImpl ();
029
030    /**
031     * The business logic for the client.
032     * @return the business logic
033     */
034    public Logic getLogic ();
035
036    /**
037     * The GUI for the client.
038     * @return the GUI.
039     */
040    public GUI getGUI ();
041
042    /**
043     * Run the client.
044     * When this method is called, it executes the client.
045     */
046    public void run ();
047
048}
049
050class ClientFactoryImpl implements ClientFactory {
051
052    public Client build () { return new ClientImpl (); }
053
054}
055
056class ClientImpl implements Client {
057
058    protected final Logic logic;
059    protected final GUI gui;
060
061    protected ClientImpl () {
062        logic = Logic.factory.build ();
063        gui = GUI.factory.build (logic);
064        Debug.out.println ("ClientImpl: Built.");
065    }
066
067    public Logic getLogic () { return logic; }
068
069    public GUI getGUI () { return gui; }
070
071    public void run () {
072        Debug.out.println 
073            ("ClientImpl.run: Starting");
074        Debug.out.println 
075            ("ClientImpl.run: Calling " + logic + ".attach (" + gui + ")");
076        logic.getSubject ().attach (gui);
077        Debug.out.println 
078            ("ClientImpl.run: Calling " + gui + ".run ()");
079        gui.run ();
080        Debug.out.println 
081            ("ClientImpl.run: Returning");
082    }
083
084}
