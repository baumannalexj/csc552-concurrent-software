001package ajeffrey.teaching.dining;
002
003import ajeffrey.teaching.debug.Debug;
004
005/**
006 * A philosopher from the dining philosophers problem.
007 * A philosopher thinks, picks up their left-hand fork,
008 * picks up their right-hand fork, then eats.
009 * Unfortunately, putting a collection of philosophers in a circle
010 * can produce deadlock, if they all pick up their lh forks before any
011 * of them have a chance to pick up their rh forks.
012 * @author Alan Jeffrey
013 * @version 1.0.1
014 */
015public interface DeadlockingPhilosopher {
016
017    /**
018     * A factory for building deadlocking philosophers.
019     */
020    public static final PhilosopherFactory factory 
021        = new DeadlockingPhilosopherFactoryImpl ();
022
023}
024
025class DeadlockingPhilosopherFactoryImpl implements PhilosopherFactory {
026
027    public Philosopher build 
028        (final Comparable lhFork, final Comparable rhFork, final String name) 
029    {
030        return new DeadlockingPhilosopherImpl (lhFork, rhFork, name);
031    }
032
033}
034
035class DeadlockingPhilosopherImpl implements Runnable, Philosopher {
036
037    final protected Object lhFork;
038    final protected Object rhFork;
039    final protected String name;
040    final protected Thread thread;
041
042    protected DeadlockingPhilosopherImpl
043        (final Object lhFork, final Object rhFork, final String name) 
044    {
045        this.lhFork = lhFork;
046        this.rhFork = rhFork;
047        this.name = name;
048        this.thread = new Thread (this);
049    }
050
051    public void start () {
052        thread.start ();
053    }
054
055    public void run () {
056        Debug.out.breakPoint (name + " is starting");
057        try {
058            while (true) {
059                Debug.out.println (name + " is thinking");
060                delay ();
061                Debug.out.println (name + " tries to pick up " + lhFork);
062                synchronized (lhFork) {
063                    Debug.out.println (name + " picked up " + lhFork);
064                    delay ();
065                    Debug.out.println (name + " tries to pick up " + rhFork);
066                    synchronized (rhFork) {
067                        Debug.out.println (name + " picked up " + rhFork);
068                        Debug.out.println (name + " starts eating");
069                        delay ();
070                        Debug.out.println (name + " finishes eating");
071                    }
072                }
073            }
074        } catch (final InterruptedException ex) {
075            Debug.out.println (name + " is interrupted");
076        }
077    }
078
079    protected void delay () throws InterruptedException {
080        Thread.currentThread().sleep ((long)(1000*Math.random()));
081    }
082
083}
