001package ajeffrey.teaching.test;
002
003public abstract class TestSoapList {
004
005    public abstract int size ();
006    public abstract Object head ();
007    public abstract TestSoapList tail ();
008    public String toString () {
009        final StringBuffer result = new StringBuffer ();
010        final int size = size ();
011        TestSoapList tmp = this;
012        for (int i=0; i < size; i++) {
013            result.append ("cons (").append (tmp.head ()).append (", ");
014            tmp = tmp.tail ();
015        }
016        if (tmp.size () == 0) {
017            result.append ("empty");
018        } else {
019            result.append ("...");
020        }
021        for (int i=0; i < size; i++) {
022            result.append (")");
023        }
024        return result.toString ();
025    }
026
027}
