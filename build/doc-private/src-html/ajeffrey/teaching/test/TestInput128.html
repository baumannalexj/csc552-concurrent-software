001package ajeffrey.teaching.test;
002
003import java.io.InputStream;
004import java.io.IOException;
005
006/**
007 * A class which reads 128 bytes from standard in, and prints it.
008 * @author Alan Jeffrey
009 * @version v1.0.1
010 */
011public class TestInput128 {
012
013    public static void main (String[] args) {
014        TestInput128 test = new TestInput128 ();
015        final byte[] input = test.get128 (System.in);
016        System.out.println ("Read: " + new String (input));
017    }
018
019    byte[] get128 (final InputStream in) {
020        final byte[] result = new byte[128];
021        try {
022            for (int bytesRead = 0; bytesRead < 128;) {
023                final int read = in.read (result, bytesRead, 128-bytesRead);
024                // Check to see if we hit the end of the stream
025                if (read > 0) {
026                    bytesRead = bytesRead + read;
027                } else {
028                    // End of stream!
029                    break;
030                }
031            }
032        } catch (IOException ex) {
033            System.err.println ("Oops: " + ex);
034        } finally {
035            return result;
036        }
037    }
038
039}
