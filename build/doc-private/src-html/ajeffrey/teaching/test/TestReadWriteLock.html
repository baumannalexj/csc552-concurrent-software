001package ajeffrey.teaching.test;
002
003import ajeffrey.teaching.util.lock.ReadWriteLock;
004
005public class TestReadWriteLock implements Runnable {
006
007    static int count = 0;
008    static Count numReaders = new Count ();
009    static Count numWriters = new Count ();
010    static Count runningThreads = new Count ();
011    final static ReadWriteLock lock = ReadWriteLock.factory.build ();
012
013    final Thread thread = new Thread (this);
014    final boolean isReader;
015
016    public TestReadWriteLock (boolean isReader) { this.isReader = isReader; }
017
018    public void start () { thread.start (); }
019
020    public void run () {
021        try {
022            if (isReader) { reader (); } else { writer (); }
023        } catch (InterruptedException ex) {
024            System.err.println ("Interrupted...");
025        }
026    }
027
028    public void reader () throws InterruptedException {
029        while (runningThreads.isPositive () || count==0) {
030            lock.acquireReadLock ();
031            numReaders.inc ();
032            delay (2);
033            if (Math.random () < 0.05) { 
034                System.out.println ("count = " + count + ", numReaders = " + numReaders + " numWriters = " + numWriters);
035            }
036            numReaders.dec ();
037            lock.releaseReadLock ();
038            delay (100);
039        }
040        System.out.println ("FINISHED: count = " + count + " (should be 100)");
041    }
042
043    public void writer () throws InterruptedException {
044        runningThreads.inc ();
045        for (int i=0; i<10; i++) {
046            lock.acquireWriteLock ();
047            numWriters.inc ();
048            int tmp = count;
049            delay (2);
050            count = tmp+1;
051            numWriters.dec ();
052            lock.releaseWriteLock ();
053            delay (2);
054        }
055        runningThreads.dec ();
056    }
057
058    public void delay (long maxDelay) throws InterruptedException {
059        thread.sleep ((long)(Math.random () * maxDelay));
060    }
061
062    public static void main (String[] args) {
063        runningThreads.inc ();
064        // Create 10 reader threads
065        for (int i=0; i<10; i++) {
066            TestReadWriteLock test = new TestReadWriteLock (true);
067            test.start ();
068        }
069        // Create 10 writer threads
070        for (int i=0; i<10; i++) {
071            TestReadWriteLock test = new TestReadWriteLock (false);
072            test.start ();
073        }
074        runningThreads.dec ();
075        // This should finish once the writers have done their job!
076    }
077
078}
079
080class Count {
081    final Object lock = new Object ();
082    int contents;
083    public String toString () { return "" + contents; }
084    void inc () { synchronized (lock) { contents++; } }
085    void dec () { synchronized (lock) { contents--; } }
086    boolean isPositive () { return contents > 0; }
087}
