012 * @author Alan Jeffrey
013 * @version 1.0.1
014 * @see Philosopher
015 */
016public class TestPhilosopher {
017
018    public static void main (String[] args) throws Exception{
019       Debug.out.addFile ("Dining.txt");
020        // Switch on step debugging
021        Debug.out.addFactory (StepDebugStream.factory);
022
023        // Switch on step debugging
024        // Debug.out.addFactory (StepDebugStream.factory);
025        // Send debugging to stderr
026        // Debug.out.addPrintStream (System.err);
027        // Create the forks
028        final Comparable fork1 = "Fork 1";
029        final Comparable fork2 = "Fork 2";
030        final Comparable fork3 = "Fork 3";
031        final Comparable fork4 = "Fork 4";
032        // Which philosopher factory to use: you may want to edit this!
033        final PhilosopherFactory factory = DeadlockingPhilosopher.factory;
034        // Create the philosophers
035        final Philosopher fred = factory.build (fork1, fork2, "Fred");
036        final Philosopher wilma = factory.build (fork2, fork3, "Wilma");
037        final Philosopher barney = factory.build (fork3, fork4, "Barney");
038        final Philosopher betty = factory.build (fork4, fork1, "Betty");
039        // Start the philosophers
040        fred.start ();
041        wilma.start ();
042        barney.start ();
043        betty.start ();
044    }
