001package ajeffrey.teaching.test;
002
003import java.io.OutputStream;
004import java.io.FileOutputStream;
005import java.io.IOException;
006
007/**
008 * A class which prints "Hello, World" the hard way to an output
009 * stream.
010 * @author Alan Jeffrey
011 * @version v1.0.1
012 */
013public class TestOutputHello {
014
015    public static void main (String[] args) {
016        TestOutputHello test = new TestOutputHello ();
017        test.printHelloWorld ("test.txt", "Hello, World\n");
018    }
019
020    void printHelloWorld (final String filename, final String msg) {
021        try {
022            final OutputStream out = new FileOutputStream (filename);
023            try {
024                for (int i=0; i < msg.length (); i++) {
025                    out.write (msg.charAt (i));
026                }
027            } catch (final IOException ex) {
028                // Some error handling code goes here
029                System.err.println ("Oops 1: " + ex);
030            } finally {
031                // Always a good idea to close streams
032                try {
033                    out.close ();
034                } catch (final IOException ex) {
035                    // Some more error handling code goes here
036                    System.err.println ("Oops 2: " + ex);
037                }
038            }
039        } catch (IOException ex) {
040            // Even more error handling code goes here
041            System.err.println ("Oops 2: " + ex);
042        }
043    }
044
045}
