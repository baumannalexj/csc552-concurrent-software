001package ajeffrey.teaching.test;
002
003import ajeffrey.teaching.util.priority.PriorityQueue;
004import ajeffrey.teaching.util.priority.PessimisticPriorityQueue;
005import ajeffrey.teaching.debug.Debug;
006import ajeffrey.teaching.debug.StepDebugStream;
007
008public class TestPriorityQueue {
009
010    public static void main (final String[] args) throws InterruptedException {
011        // Debug.out.addPrintStream (System.err);
012        // Debug.out.addFactory (StepDebugStream.factory);
013        final PriorityQueue queue = PessimisticPriorityQueue.factory.build ();
014        final Thread t1 = new Thread () { public void run () {
015            Debug.out.breakPoint ("Starting t1");
016            for (int i=0; i < args.length; i++) {
017                queue.add (args[i]);
018            }
019            Debug.out.println  ("Done t1");
020        } };
021        final Thread t2 = new Thread () { public void run () {
022            try {
023                sleep (1000);
024                Debug.out.breakPoint ("Starting t2");
025                while (t1.isAlive () || queue.size () > 0) {
026                    Debug.out.println ("Getting...");
027                    final String gotten = (String)(queue.get ());
028                    Debug.out.println ("Got: " + gotten);
029                    System.out.println ("Got: " + gotten);
030                }
031            } catch (final InterruptedException ex) {
032                Debug.out.println  ("Interrupted t2");          
033            }
034            Debug.out.println  ("Done t2");
035        } };
036        t1.start ();
037        t2.start ();
038    }
039
040}
