001package ajeffrey.teaching.test;
002
003import javax.swing.JFrame;
004import javax.swing.JScrollPane;
005import javax.swing.JTextArea;
006
007/**
008 * A class which tests Sun's JScrollPane class.
009 * It creates two threads, both of which create a new scroll pane.
010 * When run under Windows 9x, JDK1.3, it can cause a null pointer
011 * exception (there's a race condition, so it only happens intermittently),
012 * for example:
013 * <pre>
014 * java.lang.NullPointerException
015 *      at javax.swing.plaf.metal.BumpBuffer.fillBumpBuffer(Unknown Source)
016 *      at javax.swing.plaf.metal.BumpBuffer.&lt;init&gt;(Unknown Source)
017 *      at javax.swing.plaf.metal.MetalBumps.createBuffer(Unknown Source)
018 *      at javax.swing.plaf.metal.MetalBumps.setBumpColors(Unknown Source)
019 *      at javax.swing.plaf.metal.MetalBumps.&lt;init&gt;(Unknown Source)
020 *      at javax.swing.plaf.metal.MetalScrollBarUI.installDefaults(Unknown Source)
021 *      at javax.swing.plaf.basic.BasicScrollBarUI.installUI(Unknown Source)
022 *      at javax.swing.JComponent.setUI(Unknown Source)
023 *      at javax.swing.JScrollBar.updateUI(Unknown Source)
024 *      at javax.swing.JScrollBar.&lt;init&gt;(Unknown Source)
025 *      at javax.swing.JScrollBar.&lt;init&gt;(Unknown Source)
026 *      at javax.swing.JScrollPane$ScrollBar.&lt;init&gt;(Unknown Source)
027 *      at javax.swing.JScrollPane.createVerticalScrollBar(Unknown Source)
028 *      at javax.swing.JScrollPane.&lt;init&gt;(Unknown Source)
029 *      at javax.swing.JScrollPane.&lt;init&gt;(Unknown Source)
030 *      at ajeffrey.teaching.test.SimpleGUI.&lt;init&gt;(ajeffrey/teaching/test/TestJScrollPane.java:93)
031 *      at ajeffrey.teaching.test.TestJScrollPane.run(ajeffrey/teaching/test/TestJScrollPane.java:82)
032 *      at java.lang.Thread.run(Unknown Source)
033 * </pre>
034 * This violates the <i>single thread rule</i> of the Swing API,
035 * described in
036 * <a href="http://java.sun.com/products/jfc/tsc/articles/threads/threads1.html">Threads and Swing</a>:
037 * <blockquote>
038 *   Once a Swing component has been realized, all code that 
039 *   might affect or depend on the state of that component should be 
040 *   executed in the event-dispatching thread.
041 * </blockquote>
042 * In particular, the constructors for each Swing component are required to be thread-safe.
043 * Unfortunately, this isn't true of <code>JScrollPane</code>, because it relies on 
044 * <code>javax.swing.plaf.metal.MetalBumps</code>, which in turn uses a hidden 
045 * <code>BumpBuffer</code> class, which uses the following method to initialize its 
046 * <code>component</code> field to be a <code>Canvas</code> whose parent is <code>frame</code>:
047 * <pre>
048 *  protected void createComponent() {
049 *      if (frame == null) {
050 *          frame = new Frame( "bufferCreator" );
051 *      }
052 *      if (component == null ) {
053 *          component = new Canvas();
054 *          frame.add( component, BorderLayout.CENTER );
055 *      }
056 *      // fix for 4185993 (moved this outside if block)
057 *       frame.addNotify();
058 *  }
059 * </pre>
060 * Unfortunately, this has a race condition... First thread A executes createComponent(),
061 * and sets <code>frame = new Frame(...)</code> and <code>component = new Canvas ()</code>.
062 * Then before thread A gets a chance to call <code>frame.add (...)</code>, thread B
063 * calls createComponent, finds that neither <code>frame</code> nor <code>component</code>
064 * are null, and so returns, without having properly initialized <code>component</code>
065 * (it still doesn't have a parent, so calling <code>createImage</code> on it causes a
066 * <code>NullPointerException</code>.
067 * 
068 * @author <a href="http://fpl.cs.depaul.edu/ajeffrey">Alan Jeffrey</a>
069 * @version 1.0.0, 30 January 2001
070 */
071public class TestJScrollPane implements Runnable {
072
073    public static void main (String[] args) {
074        // Create two simple GUIs in two different threads, and show them.
075        Runnable task = new TestJScrollPane ();
076        Thread threadA = new Thread (task);
077        Thread threadB = new Thread (task);
078        threadA.start ();
079        threadB.start ();
080    }
081
082    public void run () { new SimpleGUI ().show (); }
083
084}
085
086class SimpleGUI implements Runnable {
087
088    // Build a simple GUI
089    protected final JFrame mainFrame = new JFrame ();
090    protected final JTextArea textArea = new JTextArea (25, 40);
091
092    // This constructor is not thread-safe:
093    protected final JScrollPane textPane = new JScrollPane (textArea);
094
095    public SimpleGUI () {
096        // A very simple GUI indeed.
097          mainFrame.getContentPane ().add (textPane);
098        mainFrame.setTitle ("Testing JScrollPane for thread safety");
099    }
100
101    public void show () {
102        // To be extra goody-goody, show () makes sure the Swing event handler
103        // thread calls pack() and setVisible(true).
104        javax.swing.SwingUtilities.invokeLater (this);
105    }
106
107    public void run () {
108        // This should only be called by the event handler thread.
109        mainFrame.pack ();
110        mainFrame.setVisible (true);
111    }
112
113}
