001package ajeffrey.teaching.https.client.logic;
002
003import ajeffrey.teaching.debug.Debug;
004
005import java.io.PrintWriter;
006
007/**
008 * The business logic for a simple HTTPS client.
009 * @author Alan Jeffrey
010 * @version 1.0.1
011 */
012public interface Logic {
013
014    /**
015     * Set the output stream for the business logic to print to.
016     * @param out the new output stream
017     */
018    public void setPrintWriter (PrintWriter out);
019
020    /**
021     * Get a URL and print the contents out to the output stream.
022     * If the URL is malformed, or the contents cannot be fetched,
023     * an error is printed to the output stream.
024     * @param url the URL to fetch
025     */
026    public void get (String url);
027
028    /**
029     * Stop getting a URL.
030     * This will stop the fetch of the URL.
031     * It is safe to call this even if there is no get running at the moment.
032     */
033    public void stop ();
034
035    /**
036     * A factory for building new business logic.
037     */
038    public static final LogicFactory factory = new LogicFactoryImpl ();
039
040}
041
042class LogicFactoryImpl implements LogicFactory {
043
044    public Logic build () { return new LogicImpl (); }
045
046}
047
048class LogicImpl implements Logic {
049
050    protected PrintWriter out;
051    protected Connection connection;
052
053    public synchronized void setPrintWriter (final PrintWriter out) { 
054        Debug.out.println ("LogicImpl.setWriter: Starting");
055        this.out = out; 
056        Debug.out.println ("LogicImpl.setWriter: Returning");
057    }
058
059    public synchronized void get (final String url) {
060        Debug.out.println ("LogicImpl.get: Starting");
061        Debug.out.println ("LogicImpl.get: Building connection");
062        connection = Connection.factory.build (url, out);
063        Debug.out.println ("LogicImpl.get: Starting connection");
064        connection.start ();
065        Debug.out.println ("LogicImpl.get: Returning");    
066    }
067
068    public synchronized void stop () {
069        Debug.out.println ("LogicImpl.stop: Starting");
070        if (connection == null) {
071            Debug.out.println ("LogicImpl.stop: connection is null");
072        } else {
073            connection.stop ();
074        }
075        Debug.out.println ("LogicImpl.stop: Returning");    
076    }
077
078}
