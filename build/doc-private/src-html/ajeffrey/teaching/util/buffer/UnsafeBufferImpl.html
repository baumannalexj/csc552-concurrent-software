001package ajeffrey.teaching.util.buffer;
002
003import ajeffrey.teaching.debug.Debug;
004
005/**
006 * An unsafe class for buffers.
007 * @author Alan Jeffrey
008 * @version 1.0.1
009 */
010
011public interface UnsafeBuffer {
012
013    /**
014     * A factory for building unsafe buffers.
015     */
016    public static BufferFactory factory = new UnsafeBufferFactoryImpl ();
017
018}
019
020class UnsafeBufferFactoryImpl implements BufferFactory {
021
022    public Buffer build (int capacity) { 
023        return new UnsafeBufferImpl (capacity);
024    }
025
026}
027
028class UnsafeBufferImpl implements Buffer {
029
030    protected final int capacity;
031    protected final Object[] contents;
032    protected int first;
033    protected int last;
034    protected int size;
035
036    UnsafeBufferImpl (final int capacity) {
037        this.capacity = capacity;
038        this.contents = new Object[capacity];
039        this.first = 0;
040        this.last = 0;
041        this.size = 0;
042    }
043    
044    protected void checkInvariants () {
045        Debug.out.println ("UnsafeBuffer.checkInvariants (): State = " + this);
046        Debug.out.assertion (size <= capacity);
047        Debug.out.assertion ((size % capacity) == ((capacity + last - first) % capacity));
048        Debug.out.assertion (capacity == contents.length);
049        for (int i=0; i < size; i++) {
050            Debug.out.assertion (contents[(first+i)%capacity] != null);
051        }
052        for (int i=size; i < capacity; i++) {
053            Debug.out.assertion (contents[(first+i)%capacity] == null);
054        }
055    }
056
057    public int capacity () { 
058        return capacity; 
059    }
060
061    public int size () { 
062        return size; 
063    }
064
065    public void put (final Object object) {
066        Debug.out.println ("UnsafeBuffer.put: Starting");
067        Debug.out.println ("UnsafeBuffer.put: Checking !full");
068        if (size == capacity) { throw new BufferFullException (); }
069        Debug.out.println ("UnsafeBuffer.put: Checking object != null");
070        if (object == null) { throw new IllegalArgumentException (); }
071        Debug.out.println ("UnsafeBuffer.put: Begin critical section.");
072        contents[last] = object;
073        Debug.out.println ("UnsafeBuffer.put: Set contents[" + last + "] = " + object);
074        final int newLast = (last + 1) % capacity;
075        Debug.out.println ("UnsafeBuffer.put: Set newLast = " + newLast);
076        last = newLast;
077        Debug.out.println ("UnsafeBuffer.put: Setting last = " + last);
078        final int newSize = size + 1;
079        Debug.out.println ("UnsafeBuffer.put: Set newSize = " + newSize);
080        size = newSize;
081        Debug.out.println ("UnsafeBuffer.put: Setting size = " + size);
082        Debug.out.println ("UnsafeBuffer.put: End critical section.");
083        checkInvariants ();
084        Debug.out.println ("UnsafeBuffer.put: Returning");
085    }
086
087    public Object get () {
088        Debug.out.println ("UnsafeBuffer.get: Starting");
089        Debug.out.println ("UnsafeBuffer.get: Checking !empty");
090        if (size == 0) { throw new BufferEmptyException (); }
091        Debug.out.println ("UnsafeBuffer.get: Begin critical section.");
092        final Object result = contents[first];
093        Debug.out.println ("UnsafeBuffer.get: Set result = " + result);
094        contents[first] = null;
095        Debug.out.println ("UnsafeBuffer.get: Set contents[" + first + "] = null");
096        final int newFirst = (first + 1) % capacity;
097        Debug.out.println ("UnsafeBuffer.get: Set newFirst = " + newFirst);
098        first = newFirst;
099        Debug.out.println ("UnsafeBuffer.get: Set first = " + first);
100        final int newSize = size - 1;
101        Debug.out.println ("UnsafeBuffer.get: Set newSize = " + newSize);
102        size = size - 1;
103        Debug.out.println ("UnsafeBuffer.get: Set size = " + size);
104        Debug.out.println ("UnsafeBuffer.get: End critical section");
105        checkInvariants ();
106        Debug.out.println ("UnsafeBuffer.get: Returning " + result);
107        return result;
108    }
109
110    public String toString () {
111        StringBuffer result = new StringBuffer 
112            ("UnsafeBuffer { capacity=" + capacity + ", size=" + size + 
113             ", first=" + first + ", last=" + last);
114        for (int i=0; i<capacity; i++) {
115            result.append (", contents[" + i + "]=" + contents[i]);
116        }
117        result.append (" }");
118        return result.toString ();
119    }
120
121}
