001package ajeffrey.teaching.util.list;
002
003import java.util.NoSuchElementException;
004import java.util.WeakHashMap;
005
006import ajeffrey.teaching.debug.Debug;
007
008/**
009 * <p>This is an implementation of immutable lists, using the 
010 * `Flyweight' pattern from the `Gang of Four' book
011 * (this technique is also called `hash consing' in the LISP world).</p>
012 * 
013 * <p>For each list <code>foo</code>, we keep a hash table of all the lists
014 * <code>bar</code> such that <code>bar.tail() == foo</code>, indexed
015 * by <code>bar.head()</code>.  When we call <code>foo.cons(hd)</code>
016 * we look up <code>hd</code>'s entry in the hash-table: if it has one,
017 * we return it <i>without building a new cons cell</i>.  If it doesn't,
018 * we build a new cons cell, put it in the hash table, and return it.</p>
019 * 
020 * <p>This all means that for any object <code>hd</code> and list 
021 * <code>tl</code>, there is at most one list <code>l</code> where
022 * <code>l.head() == hd</code> and <code>l.tail() == tl</code>.
023 * This implementation is space-efficient, but at the cost of
024 * more work when cons cells are created.</p>
025 * 
026 * @author Alan Jeffrey
027 * @version 1.0.3
028 */
029public interface FlyweightImmutableList {
030
031    /**
032     * An empty list.
033     */
034    public static final ImmutableList empty = new FlyweightEmpty ();
035
036}
037
038class FlyweightFactory {
039
040    protected static ImmutableList build 
041        (final Object hd, final ImmutableList tl, final WeakHashMap cache) 
042    {
043        Debug.out.println ("FlyweightFactory.build: Starting");
044        Debug.out.println ("FlyweightFactory.build: Cache lookup for " + hd);
045        ImmutableList result = (ImmutableList)(cache.get (hd));
046        Debug.out.println ("FlyweightFactory.build: Cache found " + result);
047        if (result == null) {
048            Debug.out.println ("FlyweightFactory.build: Cache miss");
049            Debug.out.println ("FlyweightFactory.build: Grabbing lock");
050            synchronized (cache) {
051                Debug.out.println ("FlyweightFactory.build: Grabbed lock");
052                result = (ImmutableList)(cache.get (hd));
053                Debug.out.println ("FlyweightFactory.build: Cache found " + result);
054                if (result == null) {
055                    Debug.out.println ("FlyweightFactory.build: Still a cache miss");
056                    result = new FlyweightCons (hd, tl);
057                    Debug.out.println ("FlyweightFactory.build: Adding " + result);
058                    cache.put (hd, result);
059                } else {
060                    Debug.out.println ("FlyweightFactory.build: Now a cache hit");
061                }
062                Debug.out.println ("FlyweightFactory.build: Releasing lock");
063            }
064        } else {
065            Debug.out.println ("FlyweightFactory.build: Cache hit");
066        }
067        Debug.out.println ("FlyweightFactory.build: Returning " + result);
068        return result;
069    }
070
071}
072
073class FlyweightEmpty extends ImmutableListEmpty {
074
075    protected final WeakHashMap cache = new WeakHashMap ();
076
077    public ImmutableList cons (final Object element) {
078        return FlyweightFactory.build (element, this, cache);
079    }
080
081}
082
083class FlyweightCons extends ImmutableListCons {
084
085    protected final WeakHashMap cache = new WeakHashMap ();
086
087    protected FlyweightCons (final Object hd, final ImmutableList tl) {
088        super (hd, tl);
089    }
090
091    public ImmutableList cons (final Object element) {
092        return FlyweightFactory.build (element, this, cache);
093    }
094
095}
