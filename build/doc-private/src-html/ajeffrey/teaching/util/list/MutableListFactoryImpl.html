001package ajeffrey.teaching.util.list;
002
003import java.util.NoSuchElementException;
004import ajeffrey.teaching.debug.Debug;
005
006/**
007 * An ordered collection of elements.
008 * This is a cut down version of the Java collections library,
009 * intended for teaching purposes.
010 * @author Alan Jeffrey
011 * @version 1.0.4
012 * @see MutableListFactory
013 */
014public interface MutableList {
015
016    /**
017     * Add a new element to the list.
018     * @param element the object to add
019     */
020    public void add (Object element);
021
022    /**
023     * Remove an element from the list.
024     * @param element the object to remove
025     * @exception NoSuchElementException thrown if the element 
026     *   is not in the list.
027     */
028    public void remove (Object element);
029
030    /**
031     * Get an iterator over the elements in the list.
032     * @return an iterator over the elements in the list
033     */
034    public Iterator iterator ();
035
036    /**
037     * The size of the list.
038     * @return the size of the list
039     */
040    public int size ();
041
042    /**
043     * A factory for building mutable lists.
044     */
045    public static final MutableListFactory factory = new MutableListFactoryImpl ();
046
047}
048
049class MutableListFactoryImpl implements MutableListFactory {
050
051    public MutableList build () {
052        return new MutableListImpl ();
053    }
054
055}
056
057class MutableListImpl implements MutableList {
058
059    protected ImmutableList contents = ImmutableList.empty;
060    protected final Object lock = new Object ();
061
062    public void add (final Object element) {
063        synchronized (lock) {
064            Debug.out.println 
065                ("MutableListImpl.add: Starting...");
066            final ImmutableList oldContents = contents;
067            Debug.out.println 
068                ("MutableListImpl.add: Calling " + 
069                 oldContents + ".cons (" + element + ")");
070            final ImmutableList newContents = oldContents.cons (element);
071            Debug.out.println 
072                ("MutableListImpl.add: setting contents = " +
073                 newContents);
074            contents = newContents;
075            Debug.out.println 
076                ("MutableListImpl.add: ...done.");
077        }
078    }
079
080    public void remove (final Object element) {
081        synchronized (lock) {
082            Debug.out.println 
083                ("MutableListImpl.remove: Starting...");
084            final ImmutableList oldContents = contents;
085            Debug.out.println 
086                ("MutableListImpl.remove: Calling " + oldContents + 
087                 ".remove (" + element + ")");
088            final ImmutableList newContents = oldContents.remove (element);
089            Debug.out.println 
090                ("MutableListImpl.remove: setting contents = " + 
091                 newContents);
092            contents = newContents;
093            Debug.out.println 
094                ("MutableListImpl.remove: ...done.");
095        }
096    }
097
098    public Iterator iterator () { 
099        return contents.iterator ();
100    }
101
102    public int size () {
103        return contents.size ();
104    }
105
106    public String toString () {
107        return "{ contents = " + contents + " }";
108    }
109
110}
