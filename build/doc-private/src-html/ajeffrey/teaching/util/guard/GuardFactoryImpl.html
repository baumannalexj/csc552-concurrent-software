001package ajeffrey.teaching.util.guard;
002
003import ajeffrey.teaching.debug.Debug;
004
005/**
006 * A guard class.
007 * This provides an implementation of a mutable boolean variable,
008 * plus one extra method <code>waitForTrue</code> which blocks
009 * waiting for the guard to become true.
010 * This method may return true even when the flag is false,
011 * due to transitory values.  For example, if the flag is
012 * false, and we have two threads:
013 * <ul>
014 * <li>T1 is blocked having called <code>waitForTrue</code></li>
015 * <li>T2 calls <code>setValue (true)</code> then
016 *   <code>setValue (false)</code>.<code>
017 * </ul>
018 * T1 is guaranteed to be woken up, even though once it wakes up
019 * T2 may well have set the value back to be false again.
020 * This is called <i>transitory values</i>.
021 * If T1 wants to avoid transitories, it should do something like:
022 * <pre>
023 *   while (!(guard.getValue ())) { guard.waitForTrue (); }
024 * </pre>
025 * @author Alan Jeffrey
026 * @version 1.0.1
027 */
028public interface Guard {
029
030    /**
031     * Set the value of the guard
032     * @param value the new value of the guard
033     */
034    public void setValue (boolean value);
035
036    /**
037     * Get the value of the guard
038     * @return the current value of the guard
039     */
040    public boolean getValue ();
041
042    /**
043     * Wait for the guard to become true.
044     * This method will block until the guard is set to be true.
045     */
046    public void waitForTrue () throws InterruptedException;
047
048    /**
049     * A factory for building guards.
050     */
051    public static GuardFactory factory = new GuardFactoryImpl ();
052
053}
054
055class GuardFactoryImpl implements GuardFactory {
056    
057    public Guard build (final boolean init) {
058        return new GuardImpl (init);
059    }
060
061}
062
063class GuardImpl implements Guard {
064
065    protected boolean value;
066    protected Object lock = new Object ();
067
068    protected GuardImpl (final boolean init) {
069        this.value = init;
070        Debug.out.println ("GuardImpl: built");
071    }
072
073    public void setValue (final boolean value) {
074        Debug.out.println ("GuardImpl.setValue: Starting");
075        Debug.out.println ("GuardImpl.setValue: Current state=" + this);
076        Debug.out.println ("GuardImpl.setValue: Grabbing the lock");
077        synchronized (lock) {
078            this.value = value;
079            if (value) {
080                Debug.out.println ("GuardImpl.setValue: Calling notifyAll ()");
081                lock.notifyAll ();
082            }
083            Debug.out.println ("GuardImpl.setValue: Released the lock");
084        }
085        Debug.out.println ("GuardImpl.setValue: New state=" + this);
086        Debug.out.println ("GuardImpl.setValue: Returning");
087    }
088
089    public boolean getValue () {
090        Debug.out.println ("GuardImpl.getValue: Starting");
091        final boolean result = value;
092        Debug.out.println ("GuardImpl.getValue: Returning " + result);
093        return result;
094    }
095
096    public void waitForTrue () throws InterruptedException {
097        Debug.out.println ("GuardImpl.waitForTrue: Starting");
098        if (!value) {
099            Debug.out.println ("GuardImpl.waitForTrue: The value was false");
100            Debug.out.println ("GuardImpl.waitForTrue: Grabbing the lock");
101            synchronized (lock) {
102                Debug.out.println ("GuardImpl.waitForTrue: Grabbed the lock");
103                if (!value) { 
104                    Debug.out.println ("GuardImpl.waitForTrue: Still false");
105                    Debug.out.println ("GuardImpl.waitForTrue: Waiting " + this);
106                    lock.wait (); 
107                    Debug.out.println ("GuardImpl.waitForTrue: Done waiting");
108                } else {
109                    Debug.out.println ("GuardImpl.waitForTrue: True again!");
110                }
111            }
112            Debug.out.println ("GuardImpl.waitForTrue: Released the lock");
113        } else {
114            Debug.out.println ("GuardImpl.waitForTrue: The value was true");
115        }
116        Debug.out.println ("GuardImpl.getValue: Starting");
117    }
118
119    public String toString () {
120        return "Guard { value=" + value + " }";
121    }
122
123}
