001package ajeffrey.teaching.util.guard;
002
003import ajeffrey.teaching.debug.Debug;
004
005/**
006 * A buggy guard class.
007 * <p>This class has a subtle bug with it, due to not performing
008 * a double check properly.</p>
009 * @author Alan Jeffrey
010 * @version 1.0.0
011 */
012public interface BuggyGuard {
013
014    /**
015     * A factory for building buggy guards.
016     */
017    public static GuardFactory factory = new BuggyGuardFactoryImpl ();
018
019}
020
021class BuggyGuardFactoryImpl implements GuardFactory {
022    
023    public Guard build (final boolean init) {
024        return new BuggyGuardImpl (init);
025    }
026
027}
028
029class BuggyGuardImpl implements Guard {
030
031    protected boolean value;
032    protected Object lock = new Object ();
033
034    protected BuggyGuardImpl (final boolean init) {
035        this.value = init;
036        Debug.out.println ("BuggyGuardImpl: built");
037    }
038
039    public void setValue (final boolean value) {
040        Debug.out.println ("BuggyGuardImpl.setValue: Starting");
041        Debug.out.println ("BuggyGuardImpl.setValue: Current state=" + this);
042        this.value = value;
043        if (value) {
044            Debug.out.println ("BuggyGuardImpl.setValue: Grabbing the lock");
045            synchronized (lock) { 
046                Debug.out.println ("BuggyGuardImpl.setValue: Grabbed the lock");
047                Debug.out.println ("BuggyGuardImpl.setValue: Calling notifyAll ()");
048                lock.notifyAll ();
049            }
050            Debug.out.println ("BuggyGuardImpl.setValue: Released the lock");
051        }
052        Debug.out.println ("BuggyGuardImpl.setValue: New state=" + this);
053        Debug.out.println ("BuggyGuardImpl.setValue: Returning");
054    }
055
056    public boolean getValue () {
057        Debug.out.println ("BuggyGuardImpl.getValue: Starting");
058        final boolean result = value;
059        Debug.out.println ("BuggyGuardImpl.getValue: Returning " + result);
060        return result;
061    }
062
063    public void waitForTrue () throws InterruptedException {
064        Debug.out.println ("BuggyGuardImpl.waitForTrue: Starting");
065        if (!value) {
066            Debug.out.println ("BuggyGuardImpl.waitForTrue: The value was false");
067            Debug.out.println ("BuggyGuardImpl.waitForTrue: Grabbing the lock");
068            synchronized (lock) {
069                Debug.out.println ("BuggyGuardImpl.waitForTrue: Grabbed the lock");
070                // The double-check if statement has been left out here
071                // which causes a bug!
072                Debug.out.println ("BuggyGuardImpl.waitForTrue: Waiting " + this);
073                lock.wait (); 
074                Debug.out.println ("BuggyGuardImpl.waitForTrue: Done waiting");
075            }
076            Debug.out.println ("BuggyGuardImpl.waitForTrue: Released the lock");
077        } else {
078            Debug.out.println ("BuggyGuardImpl.waitForTrue: The value was true");
079        }
080        Debug.out.println ("BuggyGuardImpl.getValue: Starting");
081    }
082
083    public String toString () {
084        return "Guard { value=" + value + " }";
085    }
086
087}
