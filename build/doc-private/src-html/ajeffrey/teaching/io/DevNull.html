001package ajeffrey.teaching.io;
002
003import java.io.OutputStream;
004import java.io.PrintStream;
005import java.io.Writer;
006import java.io.OutputStreamWriter;
007import java.io.PrintWriter;
008
009/**
010 * An output stream which discards messages
011 * @author Alan Jeffrey
012 * @version 1.0.3
013 */
014public interface DevNull {
015
016    /**
017     * An output stream which discards messages
018     */
019    public static final OutputStream out = new DevNullOutputStream ();
020
021    /**
022     * A print stream which discards messages
023     */
024    public static final PrintStream printStream = new PrintStream (out);
025
026    /**
027     * A writer which discards messages
028     */
029    public static final Writer writer = new OutputStreamWriter (out);
030
031    /**
032     * A print writer which discards messages
033     */
034    public static final PrintWriter printWriter = new PrintWriter (writer);
035
036}
037
038class DevNullOutputStream extends OutputStream {
039
040    public void write (final byte[] b) {}
041
042    public void write (final byte[] b, final int off, final int len) {}
043            
044    public void write (int b) {}
045
046}
