001package ajeffrey.teaching.io;
002
003import java.io.FilterInputStream;
004import java.io.InputStream;
005import java.io.IOException;
006import java.io.InterruptedIOException;
007
008/**
009 * This class is a hack!
010 * <p>It provides an interruptable I/O stream built on top of
011 * a non-interruptable stream.  To do this, it has to use
012 * polling under the hood, so this class will busy-wait rather
013 * than blocking.  This is very annoying, but any other solution
014 * runs into the problem that Sun removed the interruptable I/O
015 * features from Java in version 1.3 without really telling anyone:
016 * see bug #4103109, #4154947 on the Java Developers' Connection
017 * at http://developer.java.sun.com/developer/bugParade/
018 * (registration required).</p>
019 * <p>You can set the minimum and maximum delays (the stream uses
020 * exponential backoff) either by setting the default values,
021 * or in the factory's build method.</p>
022 * @author Alan Jeffrey
023 * @version 1.0.1
024 */
025public interface InterruptableIO {
026
027    /**
028     * Build a new interruptable input stream.
029     * @param in the input stream to read from
030     * @return a new interruptable input stream, which reads from in
031     */
032    public InputStream build (InputStream in);
033    
034    /**
035     * Build a new interruptable input stream.
036     * @param in the input stream to read from
037     * @param minDelay the minimum time to wait between attempts to read in ms
038     * @param maxDelay the maximum time to wait between attempts to read in ms
039     * @return a new interruptable input stream, which reads from in
040     */    
041    public InputStream build (InputStream in, int minDelay, int maxDelay);
042
043    /**
044     * Set the default delays.
045     * These are the delay parameters used by
046     * <code>build (InputStream)</code>.
047     * @param minDelay the minimum time to wait between attempts to read in ms
048     * @param maxDelay the maximum time to wait between attempts to read in ms
049     */
050    public void setDefaultDelays (int minDelay, int maxDelay);
051
052    /**
053     * An object which builds new input streams.
054     */
055    public static InterruptableIO singleton = new InterruptableIOImpl ();
056
057}
058
059class InterruptableIOImpl implements InterruptableIO {
060
061    protected int minDelay = 1;
062    protected int maxDelay = 2048;
063
064    public InputStream build (final InputStream in) {
065        return build (in, minDelay, maxDelay);
066    }
067
068    public InputStream build 
069        (final InputStream in, final int minDelay, final int maxDelay) 
070    {
071        return new InterruptableInputStream (in, minDelay, maxDelay);
072    }
073
074    public void setDefaultDelays (final int minDelay, final int maxDelay) {
075        this.minDelay = minDelay;
076        this.maxDelay = maxDelay;
077    }
078
079}
080
081class InterruptableInputStream extends FilterInputStream {
082
083    protected final int minDelay;
084    protected final int maxDelay;
085    protected int delay;
086    protected int bytesTransferred = 0;
087
088    protected InterruptableInputStream 
089        (final InputStream in, final int minDelay, final int maxDelay)
090    {
091        super (in);
092        this.minDelay = minDelay;
093        this.maxDelay = maxDelay;
094        this.delay = minDelay;
095    }
096
097    protected void pollForAvailable () throws IOException {
098        delay = delay / 2;
099        if (delay < minDelay) { delay = minDelay; }
100        try {
101            while (available () < 1) {
102                Thread.currentThread ().sleep (delay);
103                delay = delay * 2;
104                if (delay > maxDelay) { delay = maxDelay; }
105            }
106        } catch (final InterruptedException ex) {
107            InterruptedIOException toThrow = 
108                new InterruptedIOException (ex.toString ());
109            toThrow.bytesTransferred = bytesTransferred;
110            throw toThrow;
111        }
112    }
113
114    public int read (final byte[] buff, final int off, final int len) 
115    throws IOException {
116        pollForAvailable ();
117        final int result = super.read (buff, off, len);
118        if (result > 0) { bytesTransferred = bytesTransferred + result; }
119        return result;
120    }
121
122    public int read (final byte[] buff) 
123    throws IOException {
124        pollForAvailable ();
125        final int result = super.read (buff);
126        if (result > 0) { bytesTransferred = bytesTransferred + result; }
127        return result;
128    }
129
130    public int read ()
131    throws IOException {
132        pollForAvailable ();
133        final int result = super.read ();
134        if (result > 0) { bytesTransferred++; }
135        return result;
136    }
137
138}
