003/**
004 * An Observer object from the Gang of Four's Observer/Observed pattern
005 * (also known as publish/subscribe).
006 * An observer object is one which can attach itself to a 
007 * Subject object, and wait for changes to the Subject's state.
008 * When the Subject changes state, it will call the Observer's
009 * <code>update</code> method, to notify it that it should update itself.
010 * @author Alan Jeffrey
011 * @version 1.0.2
012 * @see Subject
013 */
014public interface Observer {
015
016    public void update ();
017
018}
