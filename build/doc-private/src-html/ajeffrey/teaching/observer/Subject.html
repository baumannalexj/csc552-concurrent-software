001package ajeffrey.teaching.observer;
002
003import ajeffrey.teaching.util.list.Iterator;
004import ajeffrey.teaching.util.list.MutableList;
005
006/**
007 * A Subject object from the Gang of Four's Observer/Observed pattern
008 * (also known as publish/subscribe).
009 * An subject object is one which can have Observers attached to it.
010 * When the Subject changes state, it should call the 
011 * <code>updateObservers</code> method, to notify all of the
012 * attached observers that they should update themselves.
013 * @author Alan Jeffrey
014 * @version 1.0.2
015 * @see Observer
016 */
017public interface Subject {
018
019    /**
020     * Add an observer to the list of observers observing this object.
021     * @param observer the new Observer
022     */
023    public void attach (Observer observer);
024
025    /**
026     * Remove an observer from the list of observers observing this
027     * object.
028     * @param observer the Observer object to remove
029     * @exception java.util.NoSuchElementException thrown if 
030     *   the observer is not currently observing this object
031     */
032    public void detach (Observer observer);
033
034    /**
035     * Call the update method of all of the attached observers.
036     */
037    public void updateObservers ();
038
039    /**
040     * A factory for building new subjects.
041     */
042    public static SubjectFactory factory = new SubjectFactoryImpl ();
043
044}
045
046class SubjectFactoryImpl implements SubjectFactory {
047
048    public Subject build () { return new SubjectImpl (); }
049
050}
051
052class SubjectImpl implements Subject {
053
054    protected final MutableList observers = 
055        MutableList.factory.build ();
056
057    public void attach (final Observer observer) {
058        observers.add (observer);
059    }
060
061    public void detach (final Observer observer) {
062        observers.remove (observer);
063    }
064
065    public void updateObservers () {
066        for (Iterator i = observers.iterator (); i.hasNext ();) {
067            Observer observer = (Observer)(i.next ());
068            observer.update ();
069        }
070    }
071
072    public String toString () { return "Subject { }"; }
073
074}
