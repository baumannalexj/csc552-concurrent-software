001package ajeffrey.teaching.catalog;
002
003import ajeffrey.teaching.debug.Debug;
004
005import java.awt.Frame;
006import java.awt.Panel;
007import java.awt.Label;
008import java.awt.TextField;
009import java.awt.Button;
010import java.awt.TextArea;
011import java.awt.GridLayout;
012import java.awt.BorderLayout;
013
014import java.awt.event.ActionListener;
015import java.awt.event.ActionEvent;
016
017public interface View {
018
019    public void start ();
020    public void refresh ();
021
022    public static ViewFactory factory = new ViewFactoryImpl ();
023
024}
025
026class ViewFactoryImpl implements ViewFactory {
027
028    public View build (Model model) { return new ViewImpl (model); }
029
030}
031
032class ViewImpl implements View {
033
034    final Model model;
035
036    final Frame frame = new Frame ("Catalog");
037
038    final Label keywordLabel = new Label ("Keyword:");
039    final TextField keywordField = new TextField ();
040    final Label newBookLabel = new Label ("New Book:");
041    final TextField newBookField = new TextField ();
042    final Button newBookButton = new Button ("Add Book");
043    final Button searchButton = new Button ("New Search");
044    final Button refineButton = new Button ("Refine Search");
045    final Button showButton = new Button ("Show Results");
046    final Button quitButton = new Button ("Quit");
047    final TextArea resultsArea = new TextArea ();
048
049    final ActionListener quitListener = new ActionListener () {
050            public void actionPerformed (ActionEvent ev) {
051                Debug.out.println ("Quit button pressed");
052                System.exit (0);
053            }
054        };
055
056    final ActionListener searchListener = new ActionListener () {
057            public void actionPerformed (ActionEvent ev) {
058                Debug.out.println ("Search button pressed");
059                model.search (keywordField.getText ());
060                refresh ();
061            }
062        };
063
064    final ActionListener refineListener = new ActionListener () {
065            public void actionPerformed (ActionEvent ev) {
066                Debug.out.println ("Refine button pressed");
067                model.refine (keywordField.getText ());
068                refresh ();
069            }
070        };
071
072    final ActionListener showListener = new ActionListener () {
073            public void actionPerformed (ActionEvent ev) {
074                Debug.out.println ("Show button pressed");
075                model.findTitles ();
076                refresh ();
077            }
078        };
079
080    final ActionListener newBookListener = new ActionListener () {
081            public void actionPerformed (ActionEvent ev) {
082                Debug.out.println ("NewBook button pressed");
083                model.addBook (newBookField.getText ());
084                refresh ();
085            }
086        };
087
088    ViewImpl (Model model) { this.model = model; }
089
090    public void start () {
091        searchButton.addActionListener (searchListener);
092        refineButton.addActionListener (refineListener);
093        showButton.addActionListener (showListener);
094        quitButton.addActionListener (quitListener);
095        newBookButton.addActionListener (newBookListener);
096        final Panel keywordPanel = new Panel (new BorderLayout ());
097        final Panel buttonPanel = new Panel (new GridLayout (1,4));
098        final Panel toolPanel = new Panel (new GridLayout (2,1));
099        final Panel newBookPanel = new Panel (new BorderLayout ());
100        keywordPanel.add ("West", keywordLabel);
101        keywordPanel.add ("Center", keywordField);
102        buttonPanel.add (searchButton);
103        buttonPanel.add (refineButton);
104        buttonPanel.add (showButton);
105        buttonPanel.add (quitButton);
106        toolPanel.add (keywordPanel);
107        toolPanel.add (buttonPanel);
108        newBookPanel.add ("West", newBookLabel);
109        newBookPanel.add ("Center", newBookField);
110        newBookPanel.add ("East", newBookButton);
111        refresh ();
112        frame.add ("North", toolPanel);
113        frame.add ("Center", resultsArea);
114        frame.add ("South", newBookPanel);
115        frame.setSize (500,300);
116        frame.show (); 
117    }
118
119    public void refresh () {
120        final int numTitles = model.numTitles ();
121        resultsArea.setEditable (false);
122        resultsArea.setText ("");
123        resultsArea.append ("Number of results: " + numTitles);
124        refineButton.setEnabled (numTitles > 0);
125        showButton.setEnabled (numTitles > 0);
126        if (model.hasTitles ()) {
127            Debug.out.println ("Displaying results");
128            final String[] results = model.getTitles ();
129            for (int i=0; i<results.length; i++) {
130                resultsArea.append ("\n");
131                resultsArea.append (results[i]);
132            }
133        }
134    }
135
136}
