001package ajeffrey.teaching.debug;
002
003import ajeffrey.teaching.io.JTextAreaIO;
004import ajeffrey.teaching.observer.Observer;
005
006import java.io.PrintStream;
007
008import javax.swing.JButton;
009import javax.swing.JFrame;
010import javax.swing.JPanel;
011import javax.swing.JLabel;
012import javax.swing.JScrollPane;
013import javax.swing.JTextArea;
014
015import java.awt.event.ActionEvent;
016import java.awt.event.ActionListener;
017
018import java.awt.BorderLayout;
019import java.awt.FlowLayout;
020
021/**
022 * A GUI for step debugging.
023 * @author Alan Jeffrey
024 * @version 1.0.2
025 */
026public interface StepDebugGUI extends Observer {
027
028    /**
029     * Start the GUI.
030     */
031    public void start ();
032
033    /**
034     * Stop the GUI.
035     */
036    public void stop ();
037
038    /**
039     * Get the output stream for writing to the GUI.
040     * @return an output stream which prints to the GUI.
041     */
042    public PrintStream getPrintStream ();
043
044    /**
045     * A factory for building GUIs.
046     */
047    public static StepDebugGUIFactory factory = 
048        new StepDebugGUIFactoryImpl ();
049
050}
051
052class StepDebugGUIFactoryImpl implements StepDebugGUIFactory {
053
054    // It appears that the Win9x implementation of JScrollPane
055    // is not thread safe, so we have to do some locking here.
056
057    protected final Object lock = new Object ();
058    
059    public StepDebugGUI build (final StepDebugLogic logic) {
060        synchronized (lock) {
061            return new StepDebugGUIImpl (logic);
062        }
063    }
064
065}
066
067class StepDebugGUIImpl implements StepDebugGUI {
068
069    // Get round a bug with final fields in JDK1.2
070    // protected final StepDebugLogic logic;
071    protected StepDebugLogic logic;
072
073    protected final JFrame mainFrame = new JFrame ();
074    protected final JPanel buttonPanel = new JPanel ();
075    protected final JPanel bottomPanel = new JPanel ();
076    protected final JTextArea textArea = new JTextArea (25, 40);
077    protected final JScrollPane textPanel = new JScrollPane (textArea);
078    protected final JButton runButton = new JButton ("Run");
079    protected final JButton stepButton = new JButton ("Step");
080    protected final JButton pauseButton = new JButton ("Pause");
081    protected final JLabel statusLabel = new JLabel ("Initializing...");
082
083    protected final PrintStream out = 
084        JTextAreaIO.singleton.buildPrintStream (textArea);
085
086    protected final ActionListener runActionListener =
087        new ActionListener () { public void actionPerformed (ActionEvent ev) {
088            logic.runMode ();
089        } };
090
091    protected final ActionListener stepActionListener =
092        new ActionListener () { public void actionPerformed (ActionEvent ev) {
093            logic.stepMode ();
094        } };
095
096    protected final ActionListener pauseActionListener =
097        new ActionListener () { public void actionPerformed (ActionEvent ev) {
098            logic.pauseMode ();
099        } };
100
101    protected StepDebugGUIImpl (final StepDebugLogic logic) {
102        this.logic = logic;
103    }
104
105    public void update () {
106        statusLabel.setText (logic.getModeName ());
107    }
108
109    public PrintStream getPrintStream () {
110        return out;
111    }
112
113    public void start () {
114        update ();
115        logic.attach (this);
116        bottomPanel.setLayout (new BorderLayout ());
117        buttonPanel.setLayout (new FlowLayout (FlowLayout.RIGHT));
118        textArea.setEditable (false);
119        textPanel.setHorizontalScrollBarPolicy 
120            (JScrollPane.HORIZONTAL_SCROLLBAR_ALWAYS);
121        textPanel.setVerticalScrollBarPolicy 
122            (JScrollPane.VERTICAL_SCROLLBAR_ALWAYS);
123        runButton.addActionListener (runActionListener);
124        stepButton.addActionListener (stepActionListener);
125        pauseButton.addActionListener (pauseActionListener);
126        buttonPanel.add (runButton);
127        buttonPanel.add (stepButton);
128        buttonPanel.add (pauseButton);
129        bottomPanel.add (statusLabel);
130        bottomPanel.add (BorderLayout.EAST, buttonPanel);
131        mainFrame.getContentPane ().add (textPanel);
132        mainFrame.getContentPane ().add (BorderLayout.SOUTH, bottomPanel);
133        mainFrame.setTitle ("Debugging " + logic.getThreadName ());
134        mainFrame.pack ();
135        mainFrame.setVisible (true);
136    }
137
138    public void stop () {
139        logic.detach (this);
140        mainFrame.dispose ();
141    }
142
143}
