013/**
014 * An inferface for printing debugging messages.
015 * @author Alan Jeffrey
016 * @version 1.0.5
017 */
018public interface Debug extends DebugStream {
019
020    /**
021     * A debugging output stream.
022     * This output stream will print messages including
023     * thread identifiers, to keep track of safety errors
024     * caused by concurrent programming.
025     */
026    public static final Debug out = new DebugImpl ();
027
028    /**
029     * Adds a new debug factory.
030     * This factory will be used to build new debug
031     * streams when debug messages are printed.
032     * Each thread is given its own debug stream to
033     * print to.
034     * @param factory factory to build new debugging streams
035     */
036    public void addFactory (DebugStreamFactory factory);
037
038    /**
039     * Adds a new print stream
040     * @param out an output stream to send messages to
041     */
042    public void addPrintStream (PrintStream out);
043
044    /**
045     * Adds a new output stream
046     * @param out an output stream to send messages to
047     */
048    public void addOutputStream (OutputStream out);
049
050    /**
051     * Adds a new print stream
052     * @param out an output stream to send messages to
053     */
054    public void addFile (String fileName) throws FileNotFoundException;
055
056    /**
057     * Adds a new print stream
058     * @param out an output stream to send messages to
059     * @deprecated: use addPrintStream instead
060     */
061    public void setPrintStream (PrintStream out);
062
063    /**
064     * Prints a debugging message.
065     * @param msg the message to print
066     */
067    public void println (String msg);
068
069    /**
070     * Prints a debugging message, and adds a break point
071     * where execution may pause (if using a step debugger).
072     * Note: if you do this in the main GUI thread, you may
073     * cause all GUI processing to halt!
074     * @param msg the message to print
075     */
076    public void breakPoint (String msg);
077
078    /**
079     * Makes an assertion.
080     * If the assertion is false, we add a break point.
081     * @param assertion the assertion to check.
082     */
083    public void assertion (boolean assertion);
084
085}
086
087class DebugImpl implements Debug {
088
089    final protected Object lock = new Object ();
090    final protected HashMap perThread = new HashMap ();
091    protected DebugStreamFactory[] factories = new DebugStreamFactory[0];
092
093    public void addFactory (final DebugStreamFactory factory) {
094        synchronized (lock) {
095            DebugStreamFactory[] newFactories = 
096                new DebugStreamFactory[factories.length + 1];
097            System.arraycopy (factories, 0, newFactories, 0, factories.length);
098            newFactories[factories.length] = factory;
099            factories = newFactories;
100            for (final Iterator i = perThread.values ().iterator (); i.hasNext ();) {
101                PerThread next = (PerThread)(i.next ());
102                next.setFactories (factories);
103            }
104        }       
105    }
106
107    public void addPrintStream (final PrintStream out) {
108        addFactory (new PrintDebugStreamFactory (out));
109    }
110
111    public void addOutputStream (final OutputStream out) {
112        addPrintStream (new PrintStream (out, true));
113    }
114
115    public void addFile (final String fileName) throws FileNotFoundException {
116        addOutputStream (new FileOutputStream (fileName));
117    }
118
119    public void setPrintStream (final PrintStream out) {
120        addPrintStream (out);
121    }
122    
123    protected PerThread getPerThread () {
124        final Thread current = Thread.currentThread ();
125        PerThread result = (PerThread)(perThread.get (current));
126        if (result == null) {
127            synchronized (lock) {
128                result = (PerThread)(perThread.get (current));
129                if (result == null) {
130                    result = new PerThread (factories);
131                    perThread.put (current, result);
132                }
133            }
134        }
135        return result;
136    }
137
138    public void println (final String msg) {
139        getPerThread ().println (msg);
140    }
141
142    public void breakPoint (final String msg) {
143        getPerThread ().breakPoint (msg);
144    }
145
146    public void assertion (final boolean assertion) {
147        if (!assertion) {
148            breakPoint ("FAILED ASSERTION");
149        }
150    }
151
152}
153
154class PerThread {
155
156    static int threadId = 0;
157    static synchronized int newThreadId () { return ++threadId; }
158
159    protected final String threadName = "Thread " + newThreadId ();
160    protected DebugStream[] streams = new DebugStream[0];
161    protected DebugStreamFactory[] factories;
162    protected boolean printing = true;
163
164    protected PerThread (final DebugStreamFactory[] factories) {
165        this.factories = factories;
166    }
167
168    protected void setFactories (final DebugStreamFactory[] factories) {
169        this.factories = factories;
170    }
171
172    protected void buildStreams () {
173        if (streams.length < factories.length) {
174            final boolean currentPrinting =
175                printing;
176            final DebugStream[] newStreams = 
177                new DebugStream [factories.length];
178            printing = false;
179            System.arraycopy (streams, 0, newStreams, 0, streams.length);
180            for (int i=streams.length; i < factories.length; i++) {
181                newStreams[i] = factories[i].build (threadName);
182            }
183            streams = newStreams;
184            printing = currentPrinting;         
185        }
186    }
187
188    public void println (final String msg) {
189        if (printing) {
190            buildStreams ();
191            for (int i=0; i < streams.length; i++) {
192                streams[i].println (msg);
193            }
194        }
195    }
196
197    public void breakPoint (final String msg) {
198        if (printing) {
199            buildStreams ();
200            for (int i=0; i < streams.length; i++) {
201                streams[i].breakPoint (msg);
202            }
203        }
204    }
205
206}
207
208class PrintDebugStreamFactory implements DebugStreamFactory {
209    
210    final PrintStream out;
211
212    protected PrintDebugStreamFactory (final PrintStream out) {
213        this.out = out;
214    }
215
216    public DebugStream build (final String threadName) {
217        return new PrintDebugStream (out, threadName);
218    }
219
220}
221
222class PrintDebugStream implements DebugStream {
223
224    final PrintStream out;
225    final String prefix;
226    
227    protected PrintDebugStream (final PrintStream out, final String threadName) {
228        this.out = out;
229        this.prefix = threadName + ": ";
230    }
231
232    public void println (final String msg) {
233        out.println (prefix + msg);
234    }
235
236    public void breakPoint (final String msg) {
237        out.println (prefix + msg);
238    }
