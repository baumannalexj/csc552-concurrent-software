001package ajeffrey.teaching.https.client;
002
003import ajeffrey.teaching.https.client.gui.GUI;
004import ajeffrey.teaching.https.client.logic.Logic;
005import ajeffrey.teaching.debug.Debug;
006
007import java.security.Security;
008import com.sun.net.ssl.internal.ssl.Provider;
009
010/**
011 * A simple HTTPS client.
012 * @author Alan Jeffrey
013 * @version 1.0.1
014 */
015public class Main {
016
017    public static void main (String[] args) {
018        // Switch on debugging
019        Debug.out.addPrintStream (System.err);
020        // Add Sun's SSL security provider
021        Security.addProvider(new Provider());
022        // Add Sun's https protocol handler
023        System.setProperty ("java.protocol.handler.pkgs", "com.sun.net.ssl.internal.www.protocol");
024        // Add Sun's SSL debugging
025        System.setProperty ("javax.net.debug", "ssl,handshake,data,trustmanager");
026        // Set up the business logic and GUI
027        Logic logic = Logic.factory.build ();
028        GUI gui = GUI.factory.build (logic);
029        // Kick it all off
030        gui.start ();
031    }
032
033}
