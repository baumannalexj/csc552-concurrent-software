001package ajeffrey.teaching.util.priority;
002
003import ajeffrey.teaching.util.list.ImmutableList;
004import ajeffrey.teaching.debug.Debug;
005import ajeffrey.teaching.util.time.TimeoutException;
006
007/**
008 * A pessimistic implementation of priority queues.
009 * @author Alan Jeffrey
010 * @version 1.0.1
011 */
012public interface PessimisticPriorityQueue {
013
014    /**
015     * A factory for building pessimistic priorioty queues.
016     */
017    public PriorityQueueFactory factory = new PessimisticPQFactory ();
018
019}
020
021class PessimisticPQFactory implements PriorityQueueFactory {
022
023    public PriorityQueue build () { return new PessimisticPQ (); }
024
025}
026
027class PessimisticPQ implements PriorityQueue {
028
029    protected final Object lock = new Object ();
030    protected ImmutableList contents = ImmutableList.empty;
031
032    protected ImmutableList insert (final Comparable element, final ImmutableList list) {
033        Debug.out.println ("PessimisticPQ.insert: Starting");
034        if (list.size () == 0) {
035            Debug.out.println ("PessimisticPQ.insert: List is empty");
036            Debug.out.println ("PessimisticPQ.insert: Returning");
037            return list.cons (element);
038        } else if (element.compareTo (list.head ()) <= 0) {
039            Debug.out.println ("PessimisticPQ.insert: element <= " + list.head ());
040            Debug.out.println ("PessimisticPQ.insert: Returning");
041            return list.cons (element);
042        } else {
043            Debug.out.println ("PessimisticPQ.insert: element > " + list.head ());
044            Debug.out.println ("PessimisticPQ.insert: Recursing");
045            return insert (element, list.tail ()).cons (list.head ());
046        }
047    }
048
049    public void add (final Comparable element) {
050        Debug.out.println ("PessimisticPQ.add: Starting");
051        Debug.out.println ("PessimisticPQ.add: Grabbing lock...");
052        synchronized (lock) {
053            Debug.out.println ("PessimisticPQ.add: Adding " + element + " to " + contents);
054            contents = insert (element, contents);
055            Debug.out.println ("PessimisticPQ.add: Calling notifyAll...");
056            lock.notifyAll ();
057            Debug.out.println ("PessimisticPQ.add: Releasing lock...");
058        }
059        Debug.out.println ("PessimisticPQ.add: Returning");
060    }
061
062    public Comparable get () throws InterruptedException {
063        Debug.out.println ("PessimisticPQ.get: Starting");
064        Debug.out.println ("PessimisticPQ.get: Grabbing lock...");
065        synchronized (lock) {
066            Debug.out.println ("PessimisticPQ.get: contents = " + contents);
067            while (contents.size () == 0) { 
068                Debug.out.println ("PessimisticPQ.get: Waiting...");
069                lock.wait (); 
070                Debug.out.println ("PessimisticPQ.get: Woken up");
071                Debug.out.println ("PessimisticPQ.get: contents = " + contents);
072            }
073            Debug.out.println ("PessimisticPQ.get: Calling contents.head");
074            Comparable result = (Comparable)(contents.head ());
075            contents = contents.tail ();
076            Debug.out.println ("PessimisticPQ.get: contents = " + contents);
077            Debug.out.println ("PessimisticPQ.get: Returning " + result);
078            return result;
079        }
080    }
081
082    public Comparable get (final long timeout) throws InterruptedException, TimeoutException {
083        Debug.out.println ("PessimisticPQ.get: Starting");
084        Debug.out.println ("PessimisticPQ.get: Grabbing lock...");
085        synchronized (lock) {
086            Debug.out.println ("PessimisticPQ.get: contents = " + contents);
087            final long endTime = System.currentTimeMillis () + timeout;
088            Debug.out.println ("PessimisticPQ.get: endTime = " + endTime);
089            while (contents.size () == 0) { 
090                final long delay = endTime - System.currentTimeMillis ();
091                if (delay > 0) {
092                    Debug.out.println ("PessimisticPQ.get: Waiting for " + delay);
093                    lock.wait (delay); 
094                    Debug.out.println ("PessimisticPQ.get: Woken up");
095                    Debug.out.println ("PessimisticPQ.get: contents = " + contents);
096                } else {
097                    Debug.out.println ("PessimisticPQ.get: Timeout");
098                    throw new TimeoutException ();
099                }
100            }
101            Debug.out.println ("PessimisticPQ.get: Calling contents.head");
102            Comparable result = (Comparable)(contents.head ());
103            contents = contents.tail ();
104            Debug.out.println ("PessimisticPQ.get: contents = " + contents);
105            Debug.out.println ("PessimisticPQ.get: Returning " + result);
106            return result;
107        }
108    }
109
110    public int size () {
111        return contents.size ();
112    }
113    
114}
