001package ajeffrey.teaching.util.stack;
002
003import java.util.Iterator;
004import java.util.NoSuchElementException;
005
006public interface UnsafeStack {
007
008    public void push (Object element);
009    public Object pop ();
010    public int size ();
011    public Iterator iterator ();
012  
013    public static final UnsafeStackFactory factory = new UnsafeStackFactoryImpl ();
014
015}
016
017class UnsafeStackFactoryImpl implements UnsafeStackFactory {
018
019    public UnsafeStack build () { return new UnsafeStackImpl (); }
020
021}
022
023class UnsafeStackImpl implements UnsafeStack {
024
025    protected Object[] contents = new Object[1];
026    protected int size = 0;
027
028    public void push (Object element) {
029        while (size == contents.length) { grow (); }
030        contents[size++] = element;
031    }
032    
033    public Object pop () {
034        if (size == 0) { throw new NoSuchElementException (); }
035        final Object result = contents[--size];
036        contents[size] = null;
037        return result;
038    }
039
040    public int size () { return size; }
041
042    public Iterator iterator () { return new UnsafeStackIterator (contents, size); }
043
044    protected void grow () {
045        Object[] newContents = new Object[contents.length * 2];
046        System.arraycopy (contents, 0, newContents, 0, size);
047        contents = newContents;
048    }
049
050}
051
052class UnsafeStackIterator implements Iterator {
053
054    protected final Object[] contents;
055    protected final int size;
056    protected int current = 0;
057
058    UnsafeStackIterator (final Object[] contents, final int size) { 
059        this.contents = contents; this.size = size;
060    }
061
062    public boolean hasNext () { return current < size; }
063
064    public Object next () { return contents[current++]; }
065
066    public void remove () { throw new UnsupportedOperationException (); }
067
068}
