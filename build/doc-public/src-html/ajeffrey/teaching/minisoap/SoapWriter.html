016
017public interface SoapWriter {
018   
019    public void serialize (Object obj) throws IOException;
020    public void flush () throws IOException;
021    public void close () throws IOException;
022
023    public static SoapWriterFactory factory = new SoapWriterFactoryImpl ();
024
025}
026
027class SoapWriterFactoryImpl implements SoapWriterFactory {
028
029    public SoapWriter build (final OutputStream out) {
030        try {
031            return build (new OutputStreamWriter (out, "UTF-8"));
032        } catch (final UnsupportedEncodingException ex) {
033            throw new SoapException ("This shouldn't happen " + ex);
034        }
035    }
036
037    public SoapWriter build (final Writer out) {
038        return build (new SafePrintWriter (out, "\n"));
039    }
040
041    public SoapWriter build (final SafePrintWriter out) {
042        return new SoapWriterImpl (out);
043    }
044
045}
046
047class SoapWriterImpl implements SoapWriter {
048
049    protected final SafePrintWriter out;
050    // NEW:
051    protected final WeakHashMap cache = new WeakHashMap ();
052    protected int numHandles = 0;
053    // END NEW.
054
055    protected SoapWriterImpl (final SafePrintWriter out) {
056        this.out = out;
057        Debug.out.println ("SoapWriter: Built");
058    }
059
060    public void flush () throws IOException {
061        out.flush ();
062    }
063
064    public void close () throws IOException {
065        out.close ();
066    }
067
068    public void serialize (final Object obj) throws IOException {
069        // Debug.out.println ("SoapWriter.serialize: Starting");
070        final Class objClass = obj.getClass ();
071        // Debug.out.println ("SoapWriter.serialize: objClass = " + objClass);
072        final Field[] objFields = objClass.getFields ();
073        // NEW:
074        String id = (String)(cache.get (obj));
075        // Debug.out.println ("SoapWriter.serialize: id = " + id);
076        if (id != null) {
077            out.println ("<Handle href=\"#" + id + "\"/>");
078            // Debug.out.println ("SoapWriter.serialize: Sent handle");
079            return;
080        } else {
081            id = "handle" + (++numHandles);
082            cache.put (obj, id);
083            // Debug.out.println ("SoapWriter.serialize: Added " + id + " to cache");
084        }
085        // END NEW.
086        try {
087            if (objClass.isArray ()) {
088                // Debug.out.println ("SoapWriter.serialize: array " + obj);
089                Class type = objClass.getComponentType ();
090                int length = Array.getLength (obj);
091                out.println ("<Array class=\"" + type.getName () + "\" length=\"" + length + "\" id=\"" + id + "\">");
092                for (int i=0; i<length; i++) {
093                    serialize (type, Array.get (obj, i));
094                }
095                out.println ("</Array>");
096            } else {
097                // MODIFIED TO SEND handle:
098                out.println ("<Object class=\"" + objClass.getName () + "\" id=\"" + id + "\">");
099                // REST IS UNCHANGED
100                for (int i=0; i < objFields.length; i++) {
101                    // Debug.out.println ("SoapWriter.serialize: field[" + i + "] = " + objFields[i]);
102                    if (!Modifier.isStatic (objFields[i].getModifiers ())) {
103                        final Class fieldType = objFields[i].getType ();
104                        final Object fieldContents = objFields[i].get (obj);
105                        serialize (fieldType, fieldContents);
106                    }
107                }
108                out.println ("</Object>");
109            }
110        } catch (final IllegalAccessException ex) {
111            // Debug.out.println ("SoapWriter.serialize: Caught " + ex);
112            throw new SoapException ("Caught " + ex);       
113        }
114        // Debug.out.println ("SoapWriter.serialize: Returning");
115    }
116
117    void serialize (Class type, Object obj) throws IOException {
118        if (type.isPrimitive ()) {
119            // Debug.out.println ("SoapWriter.serialize: primitive " + object);
120            out.println ("<" + type + " value=\"" + obj + "\"/>");
121        } else if (obj == null) {
122            out.println ("<Null/>");
123        }  else if (obj instanceof String) {
124            // Debug.out.println ("SoapWriter.serialize: string " + object);
125            out.println ("<String value=\"" + obj + "\"/>");
126        } else {
127            // Debug.out.println ("SoapWriter.serialize: recursive call");
128            serialize (obj);
129        }
130    }
131
132}
