001package ajeffrey.teaching.minisoap;
002
003import java.lang.reflect.Method;
004import java.lang.reflect.InvocationTargetException;
005import java.io.IOException;
006import ajeffrey.teaching.debug.Debug;
007
008public class Call implements Task {
009
010    public String objectId;
011    public String methodId;
012    public String[] argTypeIds;
013    public Object[] args;
014    public String resultId;
015
016    public Call () {} // Needed for serialization.
017
018    public Call (String objectId, String methodId, String[] argTypeIds, Object[] args, String resultId) {
019        this.objectId = objectId; 
020        this.methodId = methodId; 
021        this.argTypeIds = argTypeIds;
022        this.args = args; 
023        this.resultId = resultId;
024    }
025
026    public void run (Executor exec) {
027        try {
028            Remote object = (Remote)(Stub.lookup.get (objectId));
029            Class clazz = Class.forName (object.getRemoteInterface ());
030            Class[] argTypes = new Class[argTypeIds.length];
031            for (int i=0; i<argTypes.length; i++) {
032                argTypes[i] = Class.forName (argTypeIds[i]);
033            }
034            Method method = clazz.getDeclaredMethod (methodId, argTypes);
035            Object result = method.invoke (object, args);
036            Task returnResult = new Return (resultId, result);
037            exec.executeRemote (returnResult);
038        } catch (InvocationTargetException reflected) {
039            Throwable ex = reflected.getTargetException ();
040            // BUG: this exception should be returned to the caller!
041            Debug.out.println ("Call: failed " + ex);
042        } catch (Exception ex) {
043            // BUG: this exception should be returned to the caller!
044            Debug.out.println ("Call: failed " + ex);
045        }
046    }
047
048    public String toString () { 
049        String result = "call " + objectId + "." + methodId + " (";
050        for (int i=0; i<args.length; i++) {
051            if (i>0) { result = result + ", "; }
052            result = result + args[i]; 
053        }
054        result = result + ") to " + resultId; 
055        return result;
056    }
057
058}
