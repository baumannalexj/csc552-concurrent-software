001package ajeffrey.teaching.http.server.connection;
002
003import java.net.Socket;
004
005import java.io.InputStream;
006import java.io.FileInputStream;
007import java.io.PrintStream;
008import java.io.BufferedReader;
009import java.io.IOException;
010import java.io.FileNotFoundException;
011
012import java.net.URL;
013import java.net.MalformedURLException;
014
015import ajeffrey.teaching.io.Pipe;
016import ajeffrey.teaching.io.SocketIO;
017import ajeffrey.teaching.debug.Debug;
018
019/**
020 * A class to handle a single HTTP connection.
021 * @author Alan Jeffrey
022 * @version 1.0.1
023 */
024public interface Connection {
025
026    /**
027     * Start servicing the connection.
028     */
029    public void start ();
030
031    /**
032     * A factory for building new connections
033     */
034    public ConnectionFactory factory = new ConnectionFactoryImpl ();
035
036}
037
038class ConnectionFactoryImpl implements ConnectionFactory {
039
040    public Connection build (final Socket socket) {
041        return new ConnectionImpl (socket);
042    }
043
044}
045
046class ConnectionImpl implements Connection, Runnable {
047
048    protected final Socket socket;
049    protected final Thread thread = new Thread (this);
050
051    protected final String error400Response = 
052        "HTTP/1.0 400 Bad Request\n" +
053        "Content-Type: text/html; charset=iso-8859-1\n" +
054        "\n" +
055        "<!DOCTYPE HTML PUBLIC \"-//IETF//DTD HTML 2.0//EN\">\n" +
056        "<HTML><HEAD>\n" +
057        "<TITLE>400 Bad Request</TITLE>\n" +
058        "</HEAD><BODY>\n" +
059        "<H1>Bad Request</H1>\n" +
060        "Your browser sent a request that this server could not understand.<P>\n" +
061        "</BODY></HTML>";
062
063    protected final String error404Response =
064        "HTTP/1.0 404 Not Found\n" +
065        "Content-Type: text/html; charset=iso-8859-1\n" +
066        "\n" +
067        "<!DOCTYPE HTML PUBLIC \"-//IETF//DTD HTML 2.0//EN\">\n" +
068        "<HTML><HEAD>\n" +
069        "<TITLE>404 Not Found</TITLE>\n" +
070        "</HEAD><BODY>\n" +
071        "<H1>Not Found</H1>\n" +
072        "The requested URL was not found on this server.<P>\n" +
073        "</BODY></HTML>";
074
075    protected final String ok200Response =
076        "HTTP/1.0 200 OK";
077
078    protected ConnectionImpl (final Socket socket) {
079        this.socket = socket;
080        Debug.out.println ("ConnectionImpl: Built");
081    }
082
083    public void run () {
084        try {
085            Debug.out.println ("ConnectionImpl.run : Starting");
086            Debug.out.println ("ConnectionImpl.run : Building IO");
087            final BufferedReader in = 
088                SocketIO.singleton.buildBufferedReader (socket);
089            final PrintStream out = 
090                SocketIO.singleton.buildPrintStream (socket);
091            try {
092                Debug.out.println ("ConnectionImpl.run : Waiting for request");
093                final String request = in.readLine ();
094                Debug.out.println ("ConnectionImpl.run : Received " + request);
095                Debug.out.println ("ConnectionImpl.run : Waiting for blank line");
096                String line;
097                do {
098                    line = in.readLine ();
099                    Debug.out.println ("ConnectionImpl.run : Received " + line);
100                } while (!line.equals (""));
101                Debug.out.println ("ConnectionImpl.run : Got blank line");
102                if (request.startsWith ("GET ") && request.endsWith (" HTTP/1.0")) {
103                    Debug.out.println ("ConnectionImpl.run : Got OK request");
104                    String urlString = 
105                        request.substring (4, request.length () - 9);
106                    Debug.out.println ("ConnectionImpl.run : Parsed `" + urlString + "'");
107                    if (!urlString.startsWith ("http://")) {
108                        if (!urlString.startsWith ("/")) {
109                            urlString = "/" + urlString;
110                        }
111                        urlString = "http://localhost" + urlString;
112                    }
113                    if (urlString.endsWith ("/")) {
114                        urlString = urlString + "index.html";
115                    }
116                    String contentType = "Content-type: unknown";
117                    if (urlString.endsWith (".gif")) {
118                        contentType = "Content-type: image/gif";
119                    } else if (urlString.endsWith (".html")) {
120                        contentType = "Content-type: text/html";
121                    }
122                    final URL url = new URL (urlString);
123                    Debug.out.println ("ConnectionImpl.run : Parsed url");
124                    final String fileName = url.getFile ();
125                    Debug.out.println ("ConnectionImpl.run : Opening " + fileName);
126                    final InputStream file = new FileInputStream ("." + fileName);
127                    Debug.out.println ("ConnectionImpl.run : Printing header");
128                    out.println (ok200Response);
129                    out.println (contentType);
130                    out.println ();
131                    Debug.out.println ("ConnectionImpl.run : Building pipe");
132                    final Pipe pipe = Pipe.factory.build (file, out);
133                    Debug.out.println ("ConnectionImpl.run : Starting pipe");
134                    pipe.connect ();
135                    file.close ();
136                } else {
137                    Debug.out.println ("ConnectionImpl.run : Malformed request");
138                    out.println (error400Response);
139                }
140            } catch (final FileNotFoundException ex) {
141                Debug.out.println ("ConnectionImpl.run : Caught inner " + ex);
142                out.println (error404Response);
143            } catch (final MalformedURLException ex) {
144                Debug.out.println ("ConnectionImpl.run : Caught inner " + ex);
145                out.println (error400Response);
146            } catch (final IOException ex) {
147                Debug.out.println ("ConnectionImpl.run : Caught inner " + ex);
148                out.println (error400Response);
149            } finally {
150                Debug.out.println ("ConnectionImpl.run : Calling in.close ()");
151                in.close ();
152                Debug.out.println ("ConnectionImpl.run : Calling out.close ()");
153                out.close ();
154            }
155        } catch (final IOException ex) {
156            Debug.out.println ("ConnectionImpl.run : Caught outer " + ex);
157        } finally {
158            try {
159                Debug.out.println ("ConnectionImpl.run : Calling socket.close ()");
160                socket.close ();
161            } catch (final IOException ex) {
162                Debug.out.println ("ConnectionImpl.run : Caught closing " + ex);
163            } finally {
164                Debug.out.println ("ConnectionImpl.run : Returning");
165            }
166        }
167    }
168
169    public void start () {
170        Debug.out.println ("ConnectionImpl.start : Starting");
171        thread.start ();
172        Debug.out.println ("ConnectionImpl.start : Returning");
173    }
174
175}
