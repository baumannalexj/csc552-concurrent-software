001package ajeffrey.teaching.http.server.connection;
002
003import java.net.Socket;
004
005import java.io.InputStream;
006import java.io.FileInputStream;
007import java.io.PrintStream;
008import java.io.BufferedReader;
009import java.io.IOException;
010import java.io.FileNotFoundException;
011
012import java.net.URL;
013import java.net.MalformedURLException;
014
015import ajeffrey.teaching.http.server.parser.ParseException;
016import ajeffrey.teaching.http.server.parser.HTTPParser;
017import ajeffrey.teaching.http.server.parser.HTTPRequestHandler;
018
019import ajeffrey.teaching.io.Pipe;
020import ajeffrey.teaching.io.SocketIO;
021import ajeffrey.teaching.debug.Debug;
022
023import com.macfaq.io.SafePrintWriter;
024
025/**
026 * A class to handle a single HTTP connection.
027 * This implementation uses the JavaCC parser generator to
028 * parse incoming requests.
029 * @author Alan Jeffrey
030 * @version 1.0.2
031 */
032public interface JavaCCConnection {
033
034    /**
035     * A factory for building new connections
036     */
037    public ConnectionFactory factory = new JavaCCConnectionFactoryImpl ();
038
039}
040
041class JavaCCConnectionFactoryImpl implements ConnectionFactory {
042
043    public Connection build (final Socket socket) {
044        return new JavaCCConnectionImpl (socket);
045    }
046
047}
048
049class JavaCCConnectionImpl implements Connection, Runnable, HTTPRequestHandler {
050
051    protected final Thread thread = new Thread (this);
052    protected final Socket socket;
053    protected HTTPParser parser;
054    protected SafePrintWriter out;
055
056    protected final String error400Response = 
057        "HTTP/1.0 400 Bad Request\n" +
058        "Content-Type: text/html; charset=iso-8859-1\n" +
059        "\n" +
060        "<!DOCTYPE HTML PUBLIC \"-//IETF//DTD HTML 2.0//EN\">\n" +
061        "<HTML><HEAD>\n" +
062        "<TITLE>400 Bad Request</TITLE>\n" +
063        "</HEAD><BODY>\n" +
064        "<H1>Bad Request</H1>\n" +
065        "Your browser sent a request that this server could not understand.<P>\n" +
066        "</BODY></HTML>";
067
068    protected final String error404Response =
069        "HTTP/1.0 404 Not Found\n" +
070        "Content-Type: text/html; charset=iso-8859-1\n" +
071        "\n" +
072        "<!DOCTYPE HTML PUBLIC \"-//IETF//DTD HTML 2.0//EN\">\n" +
073        "<HTML><HEAD>\n" +
074        "<TITLE>404 Not Found</TITLE>\n" +
075        "</HEAD><BODY>\n" +
076        "<H1>Not Found</H1>\n" +
077        "The requested URL was not found on this server.<P>\n" +
078        "</BODY></HTML>";
079
080    protected final String ok200Response =
081        "HTTP/1.0 200 OK";
082
083    protected JavaCCConnectionImpl (final Socket socket) {
084        this.socket = socket;
085        Debug.out.println ("JavaCCConnectionImpl: Built");
086    }
087
088    public void handleGetRequest (final URL url) {
089        Debug.out.println ("JavaCCConnectionImpl.handleGetRequest: Starting");
090        Debug.out.println ("JavaCCConnectionImpl.handleGetRequest: url = " + url);
091        try {
092            String fileName = url.getFile ();
093            if (fileName.equals ("") || fileName.endsWith ("/")) {
094                fileName = fileName + "index.html";
095            }
096            final String contentType;
097            if (fileName.endsWith (".gif")) {
098                contentType = "Content-type: image/gif";
099            } else if (fileName.endsWith (".html")) {
100                contentType = "Content-type: text/html";
101            } else {
102                contentType = "Content-type: unknown";
103            }
104            Debug.out.println ("JavaCCConnectionImpl.handleGetRequest: Opening " + fileName);
105            final InputStream file = new FileInputStream ("." + fileName);
106            Debug.out.println ("JavaCCConnectionImpl.handleGetRequest: Printing header");
107            out.println (ok200Response);
108            out.println (contentType);
109            out.println ();
110            Debug.out.println ("JavaCCConnectionImpl.handleGetRequest: Building pipe");
111            final Pipe pipe = Pipe.factory.build (file, socket.getOutputStream ());
112            Debug.out.println ("JavaCCConnectionImpl.handleGetRequest: Starting pipe");
113            pipe.connect ();
114            file.close ();
115        } catch (final FileNotFoundException ex) {
116            try {
117                Debug.out.println ("JavaCCConnectionImpl.handleGetRequest: Caught " + ex);
118                out.println (error404Response);
119            } catch (final IOException ioex) {
120                Debug.out.println ("JavaCCConnectionImpl.handleGetRequest: Caught " + ioex);
121            }
122        } catch (final IOException ex) {
123            try {
124                Debug.out.println ("JavaCCConnectionImpl.handleGetRequest: Caught " + ex);
125                out.println (error400Response);     
126            } catch (final IOException ioex) {
127                Debug.out.println ("JavaCCConnectionImpl.handleGetRequest: Caught " + ioex);
128            }
129        } finally {
130            Debug.out.println ("JavaCCConnectionImpl.handleGetRequest: Returning");
131        }
132    }
133
134    public void run () {
135        try {
136            try {
137                Debug.out.println ("JavaCCConnectionImpl.run: Starting");
138                Debug.out.println ("JavaCCConnectionImpl.run: Building IO");
139                out = SocketIO.singleton.buildSafePrintWriter (socket, "\n");
140                parser = new HTTPParser (socket.getInputStream ());
141                Debug.out.println ("JavaCCConnectionImpl.run: Adding handler");
142                parser.addHTTPRequestHandler (this);
143                Debug.out.println ("JavaCCConnectionImpl.run: Parsing");
144                parser.HTTPRequest ();
145                Debug.out.println ("JavaCCConnectionImpl.run: Done parsing");
146            } catch (ParseException ex) {
147                Debug.out.println ("JavaCCConnectionImpl.run: Caught " + ex);
148                out.println (error400Response);     
149            }
150        } catch (final IOException ex) {
151            Debug.out.println ("JavaCCConnectionImpl.run: Caught " + ex);
152        } finally {
153            try {
154                Debug.out.println ("JavaCCConnectionImpl.run: Calling socket.close ()");
155                socket.close ();
156            } catch (final IOException ex) {
157                Debug.out.println ("JavaCCConnectionImpl.run: Caught closing " + ex);
158            } finally {
159                Debug.out.println ("JavaCCConnectionImpl.run: Returning");
160            }
161        }
162    }
163
164    public void start () {
165        Debug.out.println ("JavaCCConnectionImpl.start: Starting");
166        thread.start ();
167        Debug.out.println ("JavaCCConnectionImpl.start: Returning");
168    }
169
170}
