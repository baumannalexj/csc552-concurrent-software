011public class TestUnserializableSoap {
012
013    public static void main (String[] args) throws IOException {
014        PipedOutputStream os = new PipedOutputStream ();
015        PipedInputStream is = new PipedInputStream (os);
016        SoapWriter debug = SoapWriter.factory.build (System.out);
017        SoapWriter out = SoapWriter.factory.build (os);
018        SoapReader in = SoapReader.factory.build (is);
019        try {
020            NotReallySerializable x = new NotReallySerializable ();
021            Unserializable y = new Unserializable ();
022            x.ohDear = y;
023            debug.serialize (x);
024            debug.flush ();
025            out.serialize (x);
026            out.flush ();
027            System.out.println ("SHOULDN'T HAPPEN " + in.unserialize ());
028        } catch (SoapException ex) {
029            System.out.println ("SHOULD HAPPEN " + ex);
030            System.out.println ("SHOULD PRINT  ajeffrey.teaching.minisoap.SoapException: Trying to serialize unserializable object of type class ajeffrey.teaching.test.Unserializable");
031        }        
032    }

034}
