001package ajeffrey.teaching.test;
002
003import ajeffrey.teaching.debug.Debug;
004import ajeffrey.teaching.debug.StepDebugStream;
005
006import ajeffrey.teaching.util.guard.Guard;
007import ajeffrey.teaching.util.guard.BuggyGuard;
008
009/**
010 * A class which tests a guard class.
011 * I ran this under Sun's Linux JDK 1.3 and got:
012 * <pre>
013 * # HotSpot Virtual Machine Error, Unexpected Signal 11
014 * # Please report this error at
015 * # http://java.sun.com/cgi-bin/bugreport.cgi
016 * #
017 * # Error ID: 4F533F4C494E55580E4350500595
018 * #
019 * # Problematic Thread: prio=1 tid=0x80ba630 nid=0x5971 runnable 
020 * #
021 * </pre>
022 * Hopefully this doesn't happen too often!
023 * @author Alan Jeffrey
024 * @version 1.0.2
025 */
026public class TestGuard {
027
028    // Choose whether you want the buggy version or not!
029    // public static final Guard guard = Guard.factory.build (false);
030    public static final Guard guard = BuggyGuard.factory.build (false);
031
032    public static final Thread threadA = new Thread (new Runnable () {
033            public void run () {
034                Debug.out.breakPoint ("A starting");
035                guard.setValue (true);
036                Debug.out.println ("A done");
037            }
038        });
039
040    public static final Thread threadB = new Thread (new Runnable () {
041            public void run () {
042                Debug.out.breakPoint ("B starting");
043                try {
044                    guard.waitForTrue ();
045                } catch (final InterruptedException ex) {
046                    Debug.out.println ("B interrupted");
047                }
048                Debug.out.println ("B done");
049            }
050        });
051
052    public static void main (String[] args) throws Exception {
053        // Send debugging output to a file
054        Debug.out.addFile ("TestGuardDebug.txt");
055        // Switch on step debugging
056        Debug.out.addFactory (StepDebugStream.factory);
057        // Start the threads.
058        threadA.start ();
059        threadB.start ();
060        threadA.join ();
061        threadB.join ();
062        System.exit (0);
063    }
064
065}
