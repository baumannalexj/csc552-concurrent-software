010public class TestSoapServer {

012    public static void main (final String[] args) throws IOException {

015        while (true) {
016            final Socket socket = serverSocket.accept ();
017            final SoapReader in = SoapReader.factory.build (socket.getInputStream ());
018            final Object obj = in.unserialize ();
019            System.out.println (obj);
020            socket.close ();
021        }
022    }
