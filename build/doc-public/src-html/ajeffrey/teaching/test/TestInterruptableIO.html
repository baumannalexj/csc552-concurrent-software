003import ajeffrey.teaching.debug.Debug;
004import ajeffrey.teaching.io.InterruptableIO;

006import java.io.InputStream;
007
008/**
009 * This class tests to see whether interruptable IO has been implemented
010 * correctly.  According to the Java API, it should produce output like:
011 * <pre>
012 * Thread 1: Sleeping
013 * Thread 2: Blocking...
014 * Thread 1: Interrupting
015 * Thread 1: OK
016 * Thread 2: Thrown: InterruptedIOException ...
017 * </pre>
018 * instead, most javas will produce:
019 * <pre>
020 * Thread 1: Sleeping
021 * Thread 2: Blocking...
022 * Thread 1: Interrupting
023 * Thread 1: OK
024 * </pre>
025 * <p>ie the thread which is blocked waiting on I/O isn't interrupted
026 * properly.  This appears to be a standing issue with the JDK, and
027 * is referred to as the `interruptable I/O problem' on their
028 * developer forum.</p>
029 * <p>Sun are aware of this bug, and have declared it `unfixable'
030 * because of implementation difficulties under Win95(!).
031 * Their response is to deprectate interruptable IO, which just punts
032 * the problem back to the programmer, and produces horrible space
033 * leaks.  This is a mess!</p>
034 * <p>See bug #4103109, #4154947 on the Java Developers' Connection
035 * at http://developer.java.sun.com/developer/bugParade/
036 * (registration required).</p>
037 * @author Alan Jeffrey
038 * @version 1.0.1
039 */
040public class TestInterruptableIO {
041
042    // Pick one of the following:
043    public static InputStream test =
044        System.in;
045    //  InterruptableIO.singleton.build (System.in);
046
047    public static void main (String[] args) throws Exception {
048        Debug.out.addPrintStream (System.err);
049        Thread thread = new Thread () { public void run () {
050            try {
051                Debug.out.println ("Blocking...");
052                test.read ();
053                Debug.out.println ("Done.");
054            } catch (java.io.IOException ex) {
055                Debug.out.println ("Thrown: " + ex);
056            } catch (Exception ex) {
057                Debug.out.println ("Thrown more: " + ex);
058            }
059            } };
060        thread.start ();
061        Debug.out.println ("Sleeping");
062        Thread.sleep (2000);
063        Debug.out.println ("Interrupting");
064        thread.interrupt ();
065        Debug.out.println ("OK");
066        Thread.sleep (2000);
067        System.exit (0);
068    }
069
070}
