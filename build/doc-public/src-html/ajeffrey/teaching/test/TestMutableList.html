001package ajeffrey.teaching.test;
002
003import ajeffrey.teaching.util.list.MutableList;
004import ajeffrey.teaching.util.list.Iterator;
005import ajeffrey.teaching.debug.Debug;
006import ajeffrey.teaching.debug.StepDebugStream;
007
008public class TestMutableList {
009
010    public static void main (final String[] args) throws InterruptedException {
011        // Debug.out.addPrintStream (System.err);
012        // Debug.out.addFactory (StepDebugStream.factory);
013        final MutableList list = MutableList.factory.build ();
014        final Thread t1 = new Thread () { public void run () {
015            Debug.out.breakPoint ("Starting t1");
016            for (int i=0; i < args.length; i++) {
017                list.add (args[i]);
018            }
019            Debug.out.println  ("Done t1");
020        } };
021        final Thread t2 = new Thread () { public void run () {
022            Debug.out.breakPoint ("Starting t2");
023            while (t1.isAlive () || list.size () > 0) {
024                Debug.out.println ("Getting...");
025                for (final Iterator i = list.iterator (); i.hasNext ();) {
026                    final String gotten = (String)(i.next ());
027                    Debug.out.println ("Got: " + gotten);
028                    System.out.println ("Got: " + gotten);
029                    Debug.out.println ("Removing: " + gotten);
030                    list.remove (gotten);
031                }
032            }
033            Debug.out.println  ("Done t2");
034        } };
035        t1.start ();
036        t2.start ();
037    }
038
039}
