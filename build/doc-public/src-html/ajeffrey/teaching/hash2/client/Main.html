001package ajeffrey.teaching.hash2.client;
002
003import ajeffrey.teaching.hash2.iface.Hash;
004import ajeffrey.teaching.minisoap.Naming;
005import ajeffrey.teaching.debug.Debug;
006
007public class Main {
008
009    public static void main (String[] args) throws Exception {
010        Debug.out.addPrintStream (System.out);
011        String url;
012        if (args.length > 0) {
013            url = args[0];
014        } else {
015            url = "//localhost/hashserver";
016        }
017        final Hash hashTable = (Hash)(Naming.singleton.lookup (url));
018        final GUI gui = GUI.factory.build (hashTable);
019        gui.start ();
020    }
021
022}
